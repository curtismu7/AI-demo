--- INSERT BLOCK: admin_accounts handler ---
```js
if (action === "admin_accounts") {
  // Security: admin role guard — explicit error, never silent scope-down
  const role = req?.session?.user?.role;
  if (role !== "admin") {
    return {
      reply:
        "⚠️ This query requires admin access. Sign in as admin to view all customer accounts.",
      success: false,
      toolsCalled: ["admin_list_accounts"],
      tokensUsed: 0,
      requiresConsent: false,
      agentConfigured: true,
      tokenEvents: [],
    };
  }
  const allAccounts = await dataStore.getAllAccounts();
  const allUsers = await dataStore.getAllUsers();
  if (!allAccounts || allAccounts.length === 0) {
    return {
      reply: "No customer accounts found in the system.",
      success: true,
      toolsCalled: ["admin_list_accounts"],
      tokensUsed: 0,
      requiresConsent: false,
      agentConfigured: true,
      tokenEvents: [],
    };
  }
  const userMap = Object.fromEntries(
    (allUsers || []).map((u) => [u.id, u]),
  );
  const normalizedAccounts = allAccounts.map((a) => ({
    id: a.id,
    account_type: a.accountType,
    type: a.accountType,
    account_number: a.accountNumber,
    accountNumber: a.accountNumber,
    balance: a.balance,
    currency: a.currency || "USD",
    name: a.name,
    status: a.status || "active",
    accountHolderName: a.accountHolderName || null,
  }));
  const lines = allAccounts.map((a) => {
    const owner = (() => {
      const u = userMap[a.userId];
      if (!u) return a.userId || "unknown";
      const full = `${u.firstName || ""} ${u.lastName || ""}`.trim();
      return full || u.username || u.id;
    })();
    return `• **${owner}** — ${a.accountType} (${a.accountNumber || "—"}) — **$${Number(a.balance).toFixed(2)}** ${a.currency || "USD"}`;
  });
  return {
    reply: `All customer accounts (${allAccounts.length} total):\n\n${lines.join("\n")}`,
    success: true,
    toolsCalled: ["admin_list_accounts"],
    tokensUsed: 0,
    requiresConsent: false,
    agentConfigured: true,
    tokenEvents: [],
    accounts: normalizedAccounts,
  };
}
```

  });

    success: true,
    toolsCalled: ["admin_list_errors"],

}

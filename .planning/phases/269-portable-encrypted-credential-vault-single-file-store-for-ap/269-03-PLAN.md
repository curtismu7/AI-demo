<objective>
Wire the vault into BFF startup. This plan adds:

1. `banking_api_server/services/vaultLoader.js` — a small orchestration module that opens a vault,
   copies every entry to `configStore` via `setRaw({persist:false})`, closes the vault, and deletes
   `VAULT_PASSWORD` from `process.env`. The function is dependency-injected (configStore + vaultLib as
   parameters) so it tests cleanly without spinning up the whole BFF.

2. A short call to `loadVaultIntoConfigStore` at the very top of `banking_api_server/server.js`,
   BEFORE express() is constructed and BEFORE any route module is required. This ensures every
   subsequent `configStore.getEffective(...)` lookup sees vault-supplied values.

3. Fail-fast behavior when a vault file exists but `VAULT_PASSWORD` is unset — server logs and exits.

4. Vercel bypass: when `process.env.VERCEL === '1'`, the vault loader is skipped entirely (Vercel
   uses Encrypted Environment Variables per RESEARCH.md "Serverless treatment").

5. `.env.example` entries for `VAULT_PATH` and `VAULT_PASSWORD` (quoted per CLAUDE.md rule) with a
   no-recovery warning comment.

This plan depends on Plan 01 (vault library + configStore.setRaw `{persist:false}` extension). It
runs in Wave 2 in parallel with Plan 02 because no files overlap (Plan 02 touches scripts/vault.js;
this plan touches server.js + services/vaultLoader.js).

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| process.env ↔ vault loader | VAULT_PASSWORD enters from env; must be wiped immediately |
| vault loader ↔ configStore | Decrypted values cross this boundary into in-memory cache |
| configStore cache ↔ HTTP routes | Routes call getEffective() — must see vault values |
| host filesystem ↔ vault file | A tampered vault file is detected on open by Plan 01's HMAC |

    Step 1 — Create `banking_api_server/services/vaultLoader.js`:

    ```javascript
    'use strict';
    /**
     * Vault loader for BFF startup.
     *
     * Reads a secrets.vault file (Plan 01 lib/vault) and copies every entry into
     * configStore via setRaw(data, {persist: false}) — the in-memory cache only,
     * never SQLite. Subsequent configStore.getEffective(name) calls see the
     * vault-supplied values; restarting the process without the vault password
     * causes them to disappear (which is the security property the vault offers).
     *
     * Trust model:
     *   - Vault file is encrypted at rest; safe in `ls -la`, safe in `git diff`.
     *   - VAULT_PASSWORD env var is the auth factor; deleted immediately after open
     *     to shrink the /proc/<pid>/environ leak window (T-269-06).
     *   - KEK + DEKs are zeroed in vault.close() (T-269-08).
     *
     * Vercel: when process.env.VERCEL === '1', this loader is bypassed entirely
     * (Vercel uses Encrypted Environment Variables — see RESEARCH.md "Serverless
     * treatment" and REQ-VAULT-11).
     */
    const path = require('node:path');
    const fs = require('node:fs');

    const REPO_ROOT = path.resolve(__dirname, '..', '..');
    const DEFAULT_VAULT_PATH = path.join(REPO_ROOT, 'secrets.vault');

    async function loadVaultIntoConfigStore(opts = {}) {
      const vaultPath  = opts.vaultPath  ?? process.env.VAULT_PATH ?? DEFAULT_VAULT_PATH;
      const password   = opts.password   ?? process.env.VAULT_PASSWORD;
      const configStore = opts.configStore ?? require('./configStore');
      const vaultLib    = opts.vaultLib    ?? require('../lib/vault');
      const logger      = opts.logger      ?? console;
      const isVercel    = opts.isVercel    ?? (process.env.VERCEL === '1');

      if (isVercel) {
        logger.log('[vault] Vercel environment detected — skipping vault load (use Encrypted Environment Variables)');
        return { loaded: false, entries: 0, reason: 'vercel' };
      }

      if (!fs.existsSync(vaultPath)) {
        logger.log('[vault] no vault file at ' + vaultPath + ' — skipping (env-var + configStore values will be used)');
        return { loaded: false, entries: 0, reason: 'no_vault_file' };
      }

      if (!password) {
        const msg = '[vault] secrets.vault exists but VAULT_PASSWORD not set — refusing to start';
        logger.error(msg);
        const err = new Error(msg);
        err.code = 'VAULT_PASSWORD_MISSING';
        throw err;
      }

      let vault;
      try {
        vault = await vaultLib.openVault(vaultPath, password);
      } catch (err) {
        // Generic message only — do NOT log err.stack (might contain argon2 internal names)
        logger.error('[vault] open failed:', err.message);
        throw err;
      }

      let entryCount = 0;
      try {
        const data = {};
        for (const name of vault.list()) {
          data[name.toLowerCase()] = vault.read(name);
          entryCount++;
        }
        if (entryCount > 0) {
          await configStore.setRaw(data, { persist: false });
        }
      } finally {
        // Zero KEK + DEKs regardless of read errors. Even partial-load is acceptable;
        // the alternative (re-throw with partial cache) is worse.
        try { vault.close(); } catch (_) {}
        // Shrink the /proc/<pid>/environ leak window. Use the same env var name we sourced from.
        // (If caller passed `password` directly, this still clears the env var if it happened to
        // be set — harmless idempotent operation.)
        delete process.env.VAULT_PASSWORD;
      }

      logger.log('[vault] loaded ' + entryCount + ' entries from ' + vaultPath);
      return { loaded: true, entries: entryCount };
    }

    module.exports = { loadVaultIntoConfigStore, DEFAULT_VAULT_PATH };
    ```

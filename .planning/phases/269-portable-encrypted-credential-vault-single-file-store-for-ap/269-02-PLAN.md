<objective>
Build the operator-facing CLI for the vault: a `banking_api_server/scripts/vault.js` commander-based
multi-subcommand tool, plus 5 npm scripts in `banking_api_server/package.json` that wrap it.

Subcommands:
- `vault:get <NAME>` → prints decrypted value to stdout (only; pipe-friendly)
- `vault:set <NAME>` → prompts for value on stdin (or reads from stdin pipe); writes to vault
- `vault:list` → prints entry names, one per line
- `vault:delete <NAME>` → removes entry from vault
- `vault:rotate` → prompts for new password; re-wraps all DEKs; saves

Password supply:
- TTY interactive → `@inquirer/password` prompt (no echo)
- Non-TTY → `VAULT_PASSWORD` env var required; else fail-fast
- `set` and `rotate` print a `⚠️` "no recovery" warning per REQ-VAULT-06

Vault file discovery:
- `VAULT_PATH` env var if set
- else `path.resolve(repo-root, 'secrets.vault')` (computed via `path.join(__dirname, '..', '..', 'secrets.vault')` from `banking_api_server/scripts/`)
- If file does not exist on `vault:set` of the first entry, prompt to create it (or auto-create if `VAULT_PASSWORD` is set, non-interactive convenience for CI).
- If file does not exist on `vault:get/list/delete/rotate`, exit non-zero with VaultNotFoundError message.

Output: working CLI, 5 npm scripts, full integration test suite covering every subcommand end-to-end
via `child_process.spawnSync`.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| operator stdin ↔ CLI | Password + value enter the process via untrusted input |
| stdout ↔ shell pipe | `vault:get` value goes to potentially logged shell pipelines |
| CLI ↔ vault.js library | All crypto happens in the library; CLI is only orchestration |

    Step 3 — Create `banking_api_server/scripts/vault.js` (CommonJS, structured for testability):

    ```javascript
    #!/usr/bin/env node
    'use strict';
    const path = require('node:path');
    const fs = require('node:fs');
    const { Command } = require('commander');
    const vaultLib = require('../lib/vault');
    const {
      VaultAuthError, VaultNotFoundError, VaultEntryNotFoundError,
    } = vaultLib;

    const REPO_ROOT = path.resolve(__dirname, '..', '..');
    function resolveVaultPath() {
      return process.env.VAULT_PATH || path.join(REPO_ROOT, 'secrets.vault');
    }

    // Password supply — TTY prompt OR env var. NEVER tries to read a stored file.
    async function getPassword({ isTTY = !!process.stdin.isTTY, envPassword = process.env.VAULT_PASSWORD,
                                  promptMessage = 'Vault password:' } = {}) {
      if (envPassword) {
        if (!isTTY) {
          // Non-interactive context — log to stderr that env is being used (not stdout)
          console.error('⚠️  Using VAULT_PASSWORD from env (non-interactive mode)');
        }
        return envPassword;
      }
      if (!isTTY) {
        const err = new Error('vault password required: set VAULT_PASSWORD env or run interactively');
        err.exitCode = 1;
        throw err;
      }
      // Interactive — @inquirer/password (dynamic import; ESM module)
      const { default: passwordPrompt } = await import('@inquirer/password');
      return passwordPrompt({ message: promptMessage, mask: '*' });
    }

    function dropPasswordFromEnv() {
      // Shrink the /proc/<pid>/environ leak window — see T-269-06
      delete process.env.VAULT_PASSWORD;
    }

    function printNoRecoveryWarning() {
      console.error('⚠️  There is no password recovery. Lose this password and the vault must be');
      console.error('   rebuilt from source secrets (regenerate Helix key, worker secrets, etc.).');
    }

    // --- subcommand handlers ---

    async function cmdGet(name) {
      const password = await getPassword();
      const vault = await vaultLib.openVault(resolveVaultPath(), password);
      dropPasswordFromEnv();
      try {
        const value = vault.read(name);
        // ONLY the value goes to stdout — no banner, no log line. Pipe-friendly.
        process.stdout.write(value + '\n');
      } finally {
        vault.close();
      }
    }

    async function cmdSet(name) {
      const vaultPath = resolveVaultPath();
      const exists = fs.existsSync(vaultPath);
      printNoRecoveryWarning();
      const password = await getPassword({ promptMessage: exists ? 'Vault password:' : 'New vault password:' });
      let vault;
      if (!exists) {
        console.error('No vault at ' + vaultPath + ' — creating new vault.');
        vault = await vaultLib.createVault(vaultPath, password);
      } else {
        vault = await vaultLib.openVault(vaultPath, password);
      }
      dropPasswordFromEnv();
      try {
        // Read the value from stdin. Interactive: @inquirer/password for masking.
        // Non-interactive: read all of stdin as the value (trim trailing newline).
        let value;
        if (process.stdin.isTTY) {
          const { default: passwordPrompt } = await import('@inquirer/password');
          value = await passwordPrompt({ message: 'Value for ' + name + ':', mask: '*' });
        } else {
          value = await readAllStdin();
          // Trim exactly one trailing newline (echo "v" | npm run vault:set FOO adds \n)
          if (value.endsWith('\n')) value = value.slice(0, -1);
        }
        vault.set(name, value);
        await vault.save();
        console.error('✅ vault: set ' + name);
      } finally {
        vault.close();
      }
    }

    async function cmdList() {
      const password = await getPassword();
      const vault = await vaultLib.openVault(resolveVaultPath(), password);
      dropPasswordFromEnv();
      try {
        for (const name of vault.list()) process.stdout.write(name + '\n');
      } finally {
        vault.close();
      }
    }

    async function cmdDelete(name) {
      const password = await getPassword();
      const vault = await vaultLib.openVault(resolveVaultPath(), password);
      dropPasswordFromEnv();
      try {
        const had = vault.delete(name);
        if (!had) {
          console.error('vault: entry not found: ' + name);
          process.exitCode = 2; return;
        }
        await vault.save();
        console.error('✅ vault: deleted ' + name);
      } finally {
        vault.close();
      }
    }

    async function cmdRotate() {
      const vault = await vaultLib.openVault(resolveVaultPath(), await getPassword({ promptMessage: 'Current vault password:' }));
      dropPasswordFromEnv();
      try {
        printNoRecoveryWarning();
        // Confirm new password by asking twice.
        let newPw;
        if (process.stdin.isTTY) {
          const { default: passwordPrompt } = await import('@inquirer/password');
          newPw = await passwordPrompt({ message: 'New vault password:', mask: '*' });
          const confirm = await passwordPrompt({ message: 'Confirm new password:', mask: '*' });
          if (newPw !== confirm) {
            console.error('❌ vault: passwords do not match');
            process.exitCode = 1; return;
          }
        } else {
          newPw = process.env.VAULT_NEW_PASSWORD;
          if (!newPw) {
            console.error('vault: rotate requires VAULT_NEW_PASSWORD env in non-interactive mode');
            process.exitCode = 1; return;
          }
        }
        await vault.rotate(newPw);
        await vault.save();
        console.error('✅ vault: password rotated');
      } finally {
        vault.close();
      }
    }

    function readAllStdin() {
      return new Promise((resolve, reject) => {
        let data = '';
        process.stdin.setEncoding('utf8');
        process.stdin.on('data', (chunk) => { data += chunk; });
        process.stdin.on('end', () => resolve(data));
        process.stdin.on('error', reject);
      });
    }

    // --- dispatch (testable: parseArgsAndDispatch exported) ---

    function parseArgsAndDispatch(argv) {
      const program = new Command();
      program.name('vault').description('Encrypted credential vault CLI').exitOverride();
      program.command('get <name>').description('Print decrypted value to stdout');
      program.command('set <name>').description('Set or overwrite an entry');
      program.command('list').description('List entry names (never values)');
      program.command('delete <name>').description('Remove an entry');
      program.command('rotate').description('Rotate the vault password');
      // parseSync into a normalized {subcommand, name}
      const parsed = program.parse(['node', 'vault.js', ...argv], { from: 'user' });
      const cmd = parsed.args[0];
      const name = parsed.args[1];
      const valid = new Set(['get','set','list','delete','rotate']);
      if (!valid.has(cmd)) {
        const err = new Error('unknown subcommand: ' + cmd);
        err.exitCode = 64; throw err;
      }
      return { subcommand: cmd, name };
    }

    async function main() {
      const argv = process.argv.slice(2);
      let parsed;
      try { parsed = parseArgsAndDispatch(argv); }
      catch (err) {
        console.error('vault: ' + err.message);
        process.exit(err.exitCode || 64);
      }
      try {
        switch (parsed.subcommand) {
          case 'get':    await cmdGet(parsed.name); break;
          case 'set':    await cmdSet(parsed.name); break;
          case 'list':   await cmdList(); break;
          case 'delete': await cmdDelete(parsed.name); break;
          case 'rotate': await cmdRotate(); break;
        }
      } catch (err) {
        if (err instanceof VaultEntryNotFoundError) {
          console.error(err.message); process.exit(2);
        } else if (err instanceof VaultAuthError) {
          console.error(err.message); process.exit(3);
        } else if (err instanceof VaultNotFoundError) {
          console.error(err.message); process.exit(4);
        } else {
          console.error('vault: ' + (err.message || err)); process.exit(1);
        }
      }
    }

    // Run only if invoked directly (not when require'd by tests)
    if (require.main === module) main();

    module.exports = {
      parseArgsAndDispatch, getPassword, dropPasswordFromEnv,
      printNoRecoveryWarning, resolveVaultPath,
      cmdGet, cmdSet, cmdList, cmdDelete, cmdRotate,
    };
    ```

```javascript
// After checking act exists
if (typeof decodedToken.act !== 'object' || decodedToken.act === null) {
  return res.status(403).json({
    error: 'INSUFFICIENT_PERMISSIONS',
    error_description: 'act claim must be an object (RFC 8693 §2.2)',
    act_received: decodedToken.act,
  });

// Check that act has at least sub or client_id
if (!decodedToken.act.sub && !decodedToken.act.client_id) {

    act_received: decodedToken.act,
  });
}
```

```typescript
private decodeJwtPayload(token: string): any {
```

| T-195-02 | Spoofing | Missing act claim | Mitigate | Structural check in middleware (not object = 403); presence check (no sub/client_id = 403) |
| T-195-03 | Information Disclosure | Error messages | Mitigate | Include RFC 8693 reference in error for developer clarity; no sensitive data in error_description |
| T-195-04 | Denial of Service | Validation overhead | Accept | Structural validation is O(1); acceptable cost for security |
| T-195-05 | Elevation of Privilege | Raw token fallback | Mitigate | Fallback removed entirely; hard-fail on exchange error prevents token without delegation |

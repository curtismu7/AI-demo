```js
// banking_api_ui/src/components/education/InteractiveArchDiagram.js
import React, { useState } from 'react';
import { useTokenChainOptional } from '../../context/TokenChainContext';
import RfcLink from '../shared/RfcLink';
import './InteractiveArchDiagram.css';

// Node definitions
const NODES = {
  user:       { icon: '🧑', label: 'User / Browser', sub: 'End user', type: 'user' },
  bff:        { icon: '🖥️', label: 'OLB Application', sub: 'BFF (Express)', type: 'bff' },
  idp:        { icon: '🔐', label: 'PingOne / PF', sub: 'Authorization Server', type: 'idp' },
  agent:      { icon: '🤖', label: 'agent1', sub: 'LangChain / AI Agent', type: 'agent' },
  llm:        { icon: '🧠', label: 'LLM', sub: 'OpenAI / Bedrock', type: 'llm' },
  mcpgw:      { icon: '🔀', label: 'MCP Gateway', sub: 'mcp-gw.bxf.com :3005', type: 'mcp' },
  mcpolb:     { icon: '🏦', label: 'MCP OLB', sub: 'mcp-olb.bxf.com :8080', type: 'mcp' },
  mcpinvest:  { icon: '📈', label: 'MCP Invest', sub: 'mcp-invest.bxf.com :8081', type: 'mcp' },
  olbapi:     { icon: '📡', label: 'Banking API', sub: 'OAuth RS / OLB', type: 'api' },
  investapi:  { icon: '📡', label: 'Invest API', sub: 'OAuth RS / Invest', type: 'api' },
};

// Arrow definitions: from → to, with label and token claim annotations
const ARROWS = [
  {
    id: 'login',
    from: 'user', to: 'bff',
    label: 'PKCE Login',
    claims: { note: 'PKCE code_challenge (RFC 7636)' },
    rfc: 'RFC_7636',
  },
  {
    id: 'token_issue',
    from: 'idp', to: 'bff',
    label: 'User Token issued',
    claims: {
      aud: 'olb-resource.bxf.com',
      may_act: '{ client_id: agent1 }',
      scope: 'openid banking',
    },
    rfc: 'RFC_8693',
  },
  {
    id: 'agent_trigger',
    from: 'bff', to: 'agent',
    label: 'Agent request',
    claims: { note: 'User token forwarded to agent' },
    rfc: 'RFC_8693',
  },
  {
    id: 'llm_call',
    from: 'agent', to: 'llm',
    label: 'LLM prompt',
    claims: { note: 'Tool definitions + user query' },
    rfc: null,
  },
  {
    id: 'exchange1',
    from: 'agent', to: 'idp',
    label: 'RFC 8693 Exchange #1',
    claims: {
      subject_token: 'user_token (aud: olb-resource)',
      actor_token: 'agent CC token',
      requested_token_type: 'access_token',
      audience: 'mcp-gw.bxf.com',
    },
    rfc: 'RFC_8693',
  },
  {
    id: 'gw_token',
    from: 'idp', to: 'mcpgw',
    label: 'GW Token',
    claims: {
      aud: 'mcp-gw.bxf.com',
      act: '{ client_id: agent1 }',
      sub: 'user_sub (preserved)',
    },
    rfc: 'RFC_8693',
  },
  {
    id: 'exchange2',
    from: 'mcpgw', to: 'idp',
    label: 'RFC 8693 Exchange #2',
    claims: {
      subject_token: 'gw_token (aud: mcp-gw)',
      audience: 'mcp-olb.bxf.com',
    },
    rfc: 'RFC_8693',
  },
  {
    id: 'backend_token',
    from: 'idp', to: 'mcpolb',
    label: 'Backend Token',
    claims: {
      aud: 'mcp-olb.bxf.com',
      act: '{ client_id: agent1 }',
      sub: 'user_sub (preserved)',
    },
    rfc: 'RFC_8693',
  },
  {
    id: 'tools_list',
    from: 'agent', to: 'mcpgw',
    label: 'tools/list (MCP)',
    claims: { Authorization: 'Bearer gw_token', protocol: 'JSON-RPC 2.0' },
    rfc: 'MCP_SPEC',
  },
  {
    id: 'tools_call',
    from: 'mcpolb', to: 'olbapi',
    label: 'tools/call → API',
    claims: { Authorization: 'Bearer backend_token', scope: 'banking' },
    rfc: 'MCP_SPEC',
  },
];

function Node({ nodeKey, isActive, onClick }) {
  const n = NODES[nodeKey];
  if (!n) return null;
  return (
    <div
      className={`iad-node iad-node--${n.type}${isActive ? ' iad-node--active' : ''}`}
      onClick={() => onClick(nodeKey)}
      role="button"
      tabIndex={0}
      onKeyDown={e => e.key === 'Enter' && onClick(nodeKey)}
      title={`${n.label} — ${n.sub}`}
    >
      <div className="iad-node-icon">{n.icon}</div>
      <div className="iad-node-label">{n.label}</div>
      <div className="iad-node-sublabel">{n.sub}</div>
    </div>
  );
}

function ArrowConnector({ arrow, isActive }) {
  const [showPopup, setShowPopup] = useState(false);
  return (
    <div
      className={`iad-arrow-wrapper iad-arrow${isActive ? ' iad-arrow--active' : ''}`}
      onMouseEnter={() => setShowPopup(true)}
      onMouseLeave={() => setShowPopup(false)}
    >
      <div className="iad-arrow-line" />
      <div className="iad-arrow-head">→</div>
      <div className="iad-arrow-label">
        <div>{arrow.label}</div>
        {arrow.claims && Object.entries(arrow.claims).slice(0, 2).map(([k, v]) => (
          <div key={k} className="iad-arrow-claim"><strong>{k}:</strong> {String(v).slice(0, 30)}{String(v).length > 30 ? '…' : ''}</div>
        ))}
      </div>
      {showPopup && arrow.claims && (
        <div className="iad-claim-popup">
          {Object.entries(arrow.claims).map(([k, v]) => `${k}: ${v}`).join('\n')}
        </div>
      )}
    </div>
  );
}

export default function InteractiveArchDiagram() {
  const ctx = useTokenChainOptional();
  const events = ctx?.events || [];
  const [selectedNode, setSelectedNode] = useState(null);

  // Determine which nodes are "active" based on live token events
  const activeNodes = new Set();
  if (events.some(ev => ev.status === 'active' || ev.status === 'acquired')) {
    activeNodes.add('user');
    activeNodes.add('bff');
  }
  if (events.some(ev => ev.id?.includes('agent') || ev.id?.includes('cc'))) {
    activeNodes.add('agent');
    activeNodes.add('idp');
  }
  if (events.some(ev => ev.id?.includes('mcp'))) {
    activeNodes.add('mcpgw');
    activeNodes.add('mcpolb');
  }

  return (
    <div className="iad-root">
      <div className="iad-title">2-Token Exchange Architecture</div>
      <div className="iad-subtitle">
        Hover arrows to see token claims · Click nodes to highlight · Live token state reflected from Token Chain ·{' '}
        <RfcLink rfc="RFC_8693" />
      </div>

      {/* Main flow: horizontal */}
      <div className="iad-canvas">
        {/* Column 1: User + IDP */}
        <div className="iad-col">
          <Node nodeKey="user" isActive={activeNodes.has('user')} onClick={setSelectedNode} />
          <Node nodeKey="idp" isActive={activeNodes.has('idp')} onClick={setSelectedNode} />
        </div>

        <ArrowConnector arrow={ARROWS.find(a => a.id === 'login')} isActive={activeNodes.has('bff')} />

        {/* Column 2: BFF */}
        <div className="iad-col">
          <Node nodeKey="bff" isActive={activeNodes.has('bff')} onClick={setSelectedNode} />
        </div>

        <ArrowConnector arrow={ARROWS.find(a => a.id === 'agent_trigger')} isActive={activeNodes.has('agent')} />

        {/* Column 3: Agent + LLM */}
        <div className="iad-col">
          <Node nodeKey="agent" isActive={activeNodes.has('agent')} onClick={setSelectedNode} />
          <Node nodeKey="llm" isActive={false} onClick={setSelectedNode} />
        </div>

        <ArrowConnector arrow={ARROWS.find(a => a.id === 'tools_list')} isActive={activeNodes.has('mcpgw')} />

        {/* Column 4: MCP Gateway */}
        <div className="iad-col">
          <Node nodeKey="mcpgw" isActive={activeNodes.has('mcpgw')} onClick={setSelectedNode} />
        </div>

        <ArrowConnector arrow={ARROWS.find(a => a.id === 'tools_call')} isActive={activeNodes.has('mcpolb')} />

        {/* Column 5: MCP OLB + MCP Invest */}
        <div className="iad-col">
          <Node nodeKey="mcpolb" isActive={activeNodes.has('mcpolb')} onClick={setSelectedNode} />
          <Node nodeKey="mcpinvest" isActive={false} onClick={setSelectedNode} />
        </div>

        {/* Column 6: APIs */}
        <div className="iad-col" style={{ marginLeft: 16 }}>
          <Node nodeKey="olbapi" isActive={false} onClick={setSelectedNode} />
          <Node nodeKey="investapi" isActive={false} onClick={setSelectedNode} />
        </div>
      </div>

      {/* Token exchange detail — shows when agent/idp are active */}
      {activeNodes.has('agent') && (
        <div style={{ marginTop: 20, padding: '12px 16px', background: '#eff6ff', borderRadius: 8, fontSize: '0.8rem' }}>
          <strong>RFC 8693 Exchange Flow:</strong>{' '}
          agent1 sends user_token + agent CC token → PingOne issues GW token (aud: mcp-gw.bxf.com, act: agent1) →
          MCP Gateway re-exchanges → backend token (aud: mcp-olb.bxf.com). Subject preserved throughout.{' '}
          <RfcLink rfc="RFC_8693" section="§4" />
        </div>
      )}

      <div className="iad-legend">
        {[
          ['#60a5fa', 'User/Browser'],
          ['#34d399', 'BFF (OLB App)'],
          ['#f59e0b', 'Identity Provider'],
          ['#a78bfa', 'AI Agent'],
          ['#f472b6', 'LLM'],
          ['#2dd4bf', 'MCP Server'],
          ['#94a3b8', 'API / Resource Server'],
        ].map(([color, label]) => (
          <div key={label} className="iad-legend-item">
            <div className="iad-legend-dot" style={{ background: color }} />
            <span>{label}</span>
          </div>
        ))}
      </div>

      <div className="iad-rfc-row">
        <span>Standards:</span>
        <RfcLink rfc="RFC_8693" /> (Token Exchange) ·{' '}
        <RfcLink rfc="RFC_7636" /> (PKCE) ·{' '}
        <RfcLink rfc="MCP_SPEC" /> (Model Context Protocol) ·{' '}
        <RfcLink rfc="RFC_9728" /> (Protected Resource Metadata)
      </div>
    </div>
  );
}
```

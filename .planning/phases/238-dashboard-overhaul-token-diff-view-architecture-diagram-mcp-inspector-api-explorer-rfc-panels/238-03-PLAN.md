# Plan 238-03 — MCP Inspector Enhancement

## Goal
Add "Pair View" to `McpTrafficPage.js`: correlate `rpc_request` entries with their matching `rpc_response` entries by `correlationId`, then render them side-by-side (request JSON left, response JSON right). Also add MCP spec section links for each message type (initialize, tools/list, tools/call).

## Backend data shape (ACTUAL)
MCP traffic entries from `GET /api/mcp/traffic` each have a **single `payload` field** (not separate `request`/`response`). Direction is encoded in `entry.dir` (`'BFF→MCP'` = outbound/request, `'MCP→BFF'` = inbound/response). Entries are correlated by `entry.correlationId`.

```js
entry = {
  ts, dir, type, method, tool, statusCode, durationMs, ok, summary,
  correlationId,  // links request to its matching response
  payload: { ... } | null,  // the JSON body (request OR response, not both)
}
```

### T2 — Create McpPairView.js
**File:** `banking_api_ui/src/components/McpPairView.js`

Key design: accepts `reqEntry` (the outbound/request entry) and `respEntry` (the matching response entry, or null). Both entries have a `payload` field.

```js
// banking_api_ui/src/components/McpPairView.js
import React from 'react';
import RfcLink from './shared/RfcLink';
import './McpPairView.css';

const METHOD_DOCS = {
  'initialize':               { desc: 'MCP handshake — client sends capabilities, server responds', section: 'basic/lifecycle' },
  'notifications/initialized':{ desc: 'Client acknowledges capabilities — notification, no response expected', section: 'basic/lifecycle' },
  'tools/list':               { desc: 'Discover available MCP tools and their JSON Schema input definitions', section: 'server/tools' },
  'tools/call':               { desc: 'Invoke a named MCP tool with structured arguments', section: 'server/tools' },
  'resources/list':           { desc: 'List available MCP resources', section: 'server/resources' },
  'prompts/list':             { desc: 'List available MCP prompt templates', section: 'server/prompts' },
};

function PrettyJson({ value }) {
  if (value === null || value === undefined) return <span style={{ color: '#94a3b8' }}>—</span>;
  try {
    const str = typeof value === 'string' ? value : JSON.stringify(value, null, 2);
    return <>{str}</>;
  } catch {
    return <>{String(value)}</>;
  }
}

export default function McpPairView({ reqEntry, respEntry }) {
  if (!reqEntry) {
    return (
      <div className="mpv-root">
        <div className="mpv-empty">Select a request entry to see the request / response pair.</div>
      </div>
    );
  }

  const method = reqEntry.method;
  const docEntry = method ? METHOD_DOCS[method] : null;
  const isError = respEntry?.ok === false || respEntry?.type === 'error';
  const duration = respEntry?.durationMs ?? reqEntry?.durationMs;

  return (
    <div className="mpv-root">
      {/* Meta row */}
      <div className="mpv-meta">
        {method && <span><strong>method:</strong> {method}</span>}
        {reqEntry.tool && <span><strong>tool:</strong> {reqEntry.tool}</span>}
        {duration != null && <span><strong>duration:</strong> {duration}ms</span>}
        {reqEntry.correlationId && <span><strong>correlationId:</strong> {reqEntry.correlationId}</span>}
      </div>

      {/* Request pane — outbound payload */}
      <div className="mpv-pane">
        <div className="mpv-pane-header mpv-pane-header--req">
          <span>Request</span>
          {method && <span className="mpv-method-badge">{method}</span>}
          <span style={{ fontSize: '0.68rem', opacity: 0.75 }}>{reqEntry.dir}</span>
        </div>
        <div className="mpv-pane-body">
          <PrettyJson value={reqEntry.payload} />
        </div>
      </div>

      {/* Response pane — inbound payload */}
      <div className="mpv-pane">
        <div className={`mpv-pane-header ${isError ? 'mpv-pane-header--resp-err' : 'mpv-pane-header--resp'}`}>
          <span>{isError ? 'Error Response' : 'Response'}</span>
          {respEntry?.dir && <span style={{ fontSize: '0.68rem', opacity: 0.75 }}>{respEntry.dir}</span>}
        </div>
        <div className="mpv-pane-body">
          {respEntry
            ? <PrettyJson value={respEntry.payload} />
            : <span style={{ color: '#94a3b8' }}>No response received yet</span>}
        </div>
      </div>

      {/* RFC + description row */}
      <div className="mpv-rfc-row">
        {docEntry ? (
          <>
            <strong>{method}</strong>: {docEntry.desc} · <RfcLink rfc="MCP_SPEC" section={`§${docEntry.section}`} />
          </>
        ) : (
          <RfcLink rfc="MCP_SPEC" />
        )}
      </div>
    </div>
  );
}
```

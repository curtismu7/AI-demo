# Plan 238-04 — API Explorer Panel

## Goal
Create a new full-featured `ApiExplorerPanel` component that shows every BFF API call captured during the session in an expandable accordion. Each row shows: method, URL, status, duration. Expanding reveals full request headers/body + response body with pretty-printed JSON. Categories filter (OAuth, Token Exchange, MCP, Banking API, Admin). Educational annotations explain what each API endpoint does and which RFC governs it.

Integrate as a new "API Explorer" tab in DevToolsDashboard.

## Scope
- New: `banking_api_ui/src/components/ApiExplorerPanel.js`
- New: `banking_api_ui/src/components/ApiExplorerPanel.css`
- Modify: `banking_api_ui/src/components/DevToolsDashboard.jsx` — add "API Explorer" tab

### T2 — Create ApiExplorerPanel.js
**File:** `banking_api_ui/src/components/ApiExplorerPanel.js`

```js
// banking_api_ui/src/components/ApiExplorerPanel.js
import React, { useState, useEffect, useCallback } from 'react';
import apiClient from '../services/apiClient';
import RfcLink from './shared/RfcLink';
import './ApiExplorerPanel.css';

const CATEGORIES = ['all', 'oauth', 'exchange', 'mcp', 'banking', 'admin'];

// Educational annotations keyed by URL substring or category
const ENDPOINT_ANNOTATIONS = {
  '/api/oauth/token': { desc: 'Token endpoint: exchanges authorization code for access + id tokens (OAuth 2.0 §4.1.3)', rfc: 'RFC_6749' },
  '/api/oauth/authorize': { desc: 'Authorization endpoint: initiates PKCE flow — redirects user to PingOne login (RFC 7636)', rfc: 'RFC_7636' },
  '/api/token-exchange': { desc: 'RFC 8693 Token Exchange: swaps user token + agent CC token for MCP-scoped access token', rfc: 'RFC_8693' },
  '/api/mcp': { desc: 'MCP proxy: BFF forwards JSON-RPC 2.0 messages to MCP server with delegated token (MCP 2025-11-25)', rfc: 'MCP_SPEC' },
  '/api/transactions': { desc: 'Banking API: fetches transaction history for the authenticated user', rfc: 'RFC_6749' },
  '/api/accounts': { desc: 'Banking API: fetches account list for the authenticated user', rfc: 'RFC_6749' },
  '/api/introspect': { desc: 'Token Introspection (RFC 7662): validates token and returns active status + claims', rfc: null },
  '/api/rfc9728': { desc: 'RFC 9728 Protected Resource Metadata: discovery document for OAuth resource server', rfc: 'RFC_9728' },
};

function getAnnotation(url) {
  for (const [key, val] of Object.entries(ENDPOINT_ANNOTATIONS)) {
    if (url && url.includes(key)) return val;
  }
  return null;
}

function statusClass(status) {
  if (!status) return 'aep-status--err';
  if (status >= 200 && status < 300) return 'aep-status--ok';
  if (status >= 300 && status < 400) return 'aep-status--redirect';
  return 'aep-status--err';
}

function methodClass(method) {
  const m = (method || '').toUpperCase();
  if (m === 'GET') return 'aep-method--get';
  if (m === 'POST') return 'aep-method--post';
  if (m === 'DELETE') return 'aep-method--delete';
  return 'aep-method--put';
}

function catBadgeClass(cat) {
  return `aep-cat-badge aep-cat-badge--${cat || 'admin'}`;
}

function PrettyValue({ value }) {
  if (value === null || value === undefined) return <span style={{ color: '#94a3b8' }}>—</span>;
  const str = typeof value === 'string' ? value : JSON.stringify(value, null, 2);
  return <>{str}</>;
}

function ApiCallRow({ call }) {
  const [expanded, setExpanded] = useState(false);
  const annotation = getAnnotation(call.url);

  return (
    <div className={`aep-row${expanded ? ' aep-row--expanded' : ''}`}>
      <div
        className="aep-row-header"
        onClick={() => setExpanded(e => !e)}
        role="button"
        tabIndex={0}
        onKeyDown={e => e.key === 'Enter' && setExpanded(v => !v)}
        aria-expanded={expanded}
      >
        <span className={`aep-method ${methodClass(call.method)}`}>{(call.method || 'GET').toUpperCase()}</span>
        {call.category && (
          <span className={catBadgeClass(call.category)}>{call.category}</span>
        )}
        <span className="aep-url" title={call.url}>{call.url}</span>
        {call.response?.status && (
          <span className={`aep-status ${statusClass(call.response.status)}`}>{call.response.status}</span>
        )}
        {call.duration != null && (
          <span className="aep-duration">{call.duration}ms</span>
        )}
        <span className="aep-chevron">{expanded ? '▲' : '▼'}</span>
      </div>

      {expanded && (
        <div className="aep-detail">
          {annotation && (
            <div className="aep-annotation">
              <strong>What this does:</strong> {annotation.desc}
              {annotation.rfc && <> · <RfcLink rfc={annotation.rfc} /></>}
            </div>
          )}
          <div className="aep-detail-pane">
            <div className="aep-detail-pane-header aep-detail-pane-header--req">Request</div>
            <div className="aep-detail-body">
              <PrettyValue value={call.request} />
            </div>
          </div>
          <div className="aep-detail-pane">
            <div className="aep-detail-pane-header aep-detail-pane-header--resp">Response</div>
            <div className="aep-detail-body">
              <PrettyValue value={call.response} />
            </div>
          </div>
        </div>
      )}
    </div>
  );
}

export default function ApiExplorerPanel() {
  const [calls, setCalls] = useState([]);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState(null);
  const [activeFilter, setActiveFilter] = useState('all');

  const fetchCalls = useCallback(async () => {
    setLoading(true);
    setError(null);
    try {
      const { data } = await apiClient.get('/api/api-calls?sessionId=dashboard&limit=100');
      if (data.success) {
        setCalls(data.calls || []);
      } else {
        setError(data.error || 'Failed to fetch API calls');
      }
    } catch (err) {
      setError(err.message || 'Failed to fetch API calls');
    } finally {
      setLoading(false);
    }
  }, []);

  useEffect(() => {
    fetchCalls();
    const interval = setInterval(fetchCalls, 5000);
    return () => clearInterval(interval);
  }, [fetchCalls]);

  const filtered = activeFilter === 'all'
    ? calls
    : calls.filter(c => c.category === activeFilter);

  return (
    <div className="aep-root">
      <div className="aep-toolbar">
        <span className="aep-title">API Explorer</span>
        {CATEGORIES.map(cat => (
          <button
            key={cat}
            type="button"
            className={`aep-filter-btn${activeFilter === cat ? ' aep-filter-btn--active' : ''}`}
            onClick={() => setActiveFilter(cat)}
          >
            {cat === 'all' ? 'All' : cat.charAt(0).toUpperCase() + cat.slice(1)}
          </button>
        ))}
        <button type="button" className="aep-refresh-btn" onClick={fetchCalls}>
          ↺ Refresh
        </button>
      </div>

      <div className="aep-list">
        {loading && calls.length === 0 && <div className="aep-loading">Loading API calls…</div>}
        {error && <div className="aep-empty" style={{ color: '#991b1b' }}>{error}</div>}
        {!loading && !error && filtered.length === 0 && (
          <div className="aep-empty">
            No {activeFilter !== 'all' ? `"${activeFilter}" ` : ''}API calls recorded yet.
            {activeFilter === 'all' && ' Interact with the app — OAuth, banking, or AI agent actions — to see calls appear here.'}
          </div>
        )}
        {filtered.map(call => (
          <ApiCallRow key={call.id || `${call.url}-${call.timestamp}`} call={call} />
        ))}
      </div>
    </div>
  );
}
```

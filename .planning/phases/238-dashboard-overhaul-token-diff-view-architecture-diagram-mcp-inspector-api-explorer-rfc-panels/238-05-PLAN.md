### T2 — Create NarrativePanel.js
**File:** `banking_api_ui/src/components/NarrativePanel.js`

```js
// banking_api_ui/src/components/NarrativePanel.js
import React from 'react';
import { useTokenChainOptional } from '../context/TokenChainContext';
import RfcLink from './shared/RfcLink';
import './NarrativePanel.css';

// Derive current flow step from token events
function deriveStep(events) {
  if (!events || events.length === 0) return 'idle';
  const hasActive = events.some(ev => ev.status === 'active' || ev.status === 'acquired');
  const hasExchanging = events.some(ev => ev.status === 'acquiring');
  const hasMcp = events.some(ev => ev.id?.includes('mcp') && ev.status === 'acquired');
  const hasAgent = events.some(ev => ev.id?.includes('agent') || ev.id?.includes('cc'));
  const hasFailed = events.some(ev => ev.status === 'failed');
  if (hasFailed) return 'error';
  if (hasMcp) return 'mcp';
  if (hasExchanging) return 'exchanging';
  if (hasAgent) return 'exchange';
  if (hasActive) return 'authed';
  return 'idle';
}

const STEPS = [
  {
    num: 1,
    id: 'authed',
    title: 'User Authenticated',
    content: (events) => {
      const userEv = events.find(ev => ev.id === 'user_token' || ev.id?.includes('user'));
      const aud = userEv?.claims?.aud;
      return (
        <>
          User logged in via PKCE OAuth flow. BFF received an access token from PingOne.{' '}
          {aud && <>Audience: <span className="np-claim-highlight">aud: {Array.isArray(aud) ? aud[0] : aud}</span>{' '}</>}
          The token contains a <span className="np-claim-highlight">may_act</span> claim pre-authorizing the AI agent to perform a token exchange.{' '}
          <RfcLink rfc="RFC_7636" section="§4" />
        </>
      );
    },
    activeOn: ['authed', 'exchange', 'exchanging', 'mcp'],
  },
  {
    num: 2,
    id: 'exchange',
    title: 'Agent CC Token Acquired',
    content: () => (
      <>
        The AI agent obtained its own Client Credentials token from PingOne.
        This will be sent as the <span className="np-claim-highlight">actor_token</span> in the RFC 8693 exchange,
        establishing the delegation chain (<span className="np-claim-highlight">act</span> claim).{' '}
        <RfcLink rfc="RFC_8693" section="§4.1" />
      </>
    ),
    activeOn: ['exchange', 'exchanging', 'mcp'],
  },
  {
    num: 3,
    id: 'exchanging',
    title: 'RFC 8693 Token Exchange in Progress',
    content: () => (
      <>
        agent1 is sending both tokens to PingOne's token endpoint with{' '}
        <span className="np-claim-highlight">grant_type=urn:ietf:params:oauth:grant-type:token-exchange</span>.
        PingOne will issue a new token scoped to the MCP Gateway audience (
        <span className="np-claim-highlight">aud: mcp-gw.ping.demo</span>),
        preserving the user's <span className="np-claim-highlight">sub</span> and
        adding the <span className="np-claim-highlight">act</span> delegation claim.{' '}
        <RfcLink rfc="RFC_8693" section="§3" />
      </>
    ),
    activeOn: ['exchanging'],
  },
  {
    num: 4,
    id: 'mcp',
    title: 'MCP Token Active — Tools Available',
    content: (events) => {
      const mcpEv = events.find(ev => ev.id?.includes('mcp') && ev.status === 'acquired');
      const aud = mcpEv?.claims?.aud;
      const act = mcpEv?.claims?.act;
      return (
        <>
          Exchange complete. MCP access token issued with{' '}
          {aud && <><span className="np-claim-highlight">aud: {Array.isArray(aud) ? aud[0] : aud}</span>{' '}</>}
          {act && <><span className="np-claim-highlight">act: {JSON.stringify(act)}</span>{' '}</>}
          The agent can now call MCP tools (tools/list, tools/call) on the MCP Gateway.
          The MCP Gateway will perform a second exchange to narrow the token to the backend API audience.{' '}
          <RfcLink rfc="MCP_SPEC" section="§server/tools" />
        </>
      );
    },
    activeOn: ['mcp'],
  },
];

export default function NarrativePanel() {
  const ctx = useTokenChainOptional();
  const events = ctx?.events || [];
  const currentStep = deriveStep(events);
  const isLive = currentStep !== 'idle';

  return (
    <div className="np-root">
      <div className="np-title">
        What's Happening Now
        {isLive && <span className="np-live-badge">LIVE</span>}
      </div>
      <div className="np-subtitle">
        Contextual explanation of the current OAuth + agent + MCP flow state.
        Trigger the AI Agent to see each step activate.
      </div>

      {currentStep === 'idle' && (
        <div className="np-card np-card--idle">
          <div style={{ fontSize: '0.83rem', color: '#64748b' }}>
            No active token exchange yet. Log in and use the AI Agent to trigger the 2-Token Exchange flow.
            You'll see each step below light up as the tokens are acquired and exchanged.
          </div>
        </div>
      )}

      {currentStep === 'error' && (
        <div className="np-card np-card--error">
          <div style={{ fontSize: '0.83rem', color: '#991b1b' }}>
            <strong>Exchange failed.</strong> Check the Token Chain panel for details.
            Common causes: expired user token, misconfigured may_act claim, or MCP server unavailable.
          </div>
        </div>
      )}

      {STEPS.map(step => {
        const isActive = step.activeOn.includes(currentStep);
        const isCurrent = step.id === currentStep || (currentStep === 'exchanging' && step.id === 'exchanging');
        return (
          <div
            key={step.id}
            className={`np-step-row${!isActive ? ' np-step-inactive' : ''}`}
          >
            <span className="np-step-num">{step.num}</span>
            <div className="np-step-content">
              <strong>{step.title}</strong>
              {isActive && isCurrent && (
                <div style={{ marginTop: 4 }}>
                  {step.content(events)}
                </div>
              )}
              {isActive && !isCurrent && (
                <div style={{ marginTop: 2, fontSize: '0.75rem', color: '#6b7280' }}>Complete ✓</div>
              )}
            </div>
          </div>
        );
      })}

      <hr className="np-divider" />
      <div style={{ fontSize: '0.73rem', color: '#94a3b8', lineHeight: 1.6 }}>
        Switch to the <strong>Token Diff</strong> tab to see exactly which JWT claims change at each hop ·{' '}
        <strong>MCP Traffic</strong> to see JSON-RPC messages ·{' '}
        <strong>API Explorer</strong> to see every BFF API call ·{' '}
        <RfcLink rfc="RFC_8693" />
      </div>
    </div>
  );
}
```

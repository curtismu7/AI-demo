```javascript
// generateTestToken(options: {sub, aud, scope[], act?, exp?, expiresIn?})
// Returns: signed JWT string (for testing; not signed with real key)

function generateTestToken(options = {}) {
  const payload = {
    sub: options.sub || "test-user-123",
    aud: options.aud || "https://banking-resource-server.banking-demo.com",
    scope: (options.scope || []).join(" "),
    iat: Math.floor(Date.now() / 1000),
    exp: options.exp ?? Math.floor(Date.now() / 1000) + (options.expiresIn || 3600),
    client_id: "test-client"
  };
  
  if (options.act) payload.act = options.act; // RFC 8693 delegation
  
  return jwt.sign(payload, "test-secret-not-used-for-verification", { 
    algorithm: "HS256",
    noTimestamp: false 
  });
}

// Scenario-specific generators:
// - generateWrongScopeToken() → token with ["profile", "email"] (no mcp: scopes)
// - generateWrongAudToken() → token with aud="https://some-other-service.com"
// - generateMissingActToken() → token without act claim
// - generateAgentToken() → token with only "banking:agent" scope
// - generateExpiredToken() → token with exp time in the past
```

- act claim structure: `{client_id: "mcp-agent", sub: "user-123"}` (RFC 8693 compliant)
- Timestamp handling: iat (issued at) = now, exp = configurable (default +1 hour, or set to past time for expiration tests)

**Implementation Structure:**

```javascript
// banking_api_server/routes/testTokenScenarios.js

const express = require("express");
const router = express.Router();
const { generateTestToken, generateWrongScopeToken, ... } = require("../middleware/testTokenGenerator");

// Feature flag check
router.use((req, res, next) => {
  if (process.env.FF_TEST_TOKEN_SCENARIOS !== "true" && process.env.NODE_ENV === "production") {
    return res.status(403).json({ error: "Test routes disabled in production" });
  }
  next();
});

// Catch-all for test scenarios
router.post("/scenario/:scenarioId", async (req, res) => {
  const { scenarioId } = req.params;
  
  try {
    const result = await handleTestScenario(scenarioId);
    res.json(result);
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

async function handleTestScenario(scenarioId) {
  switch (scenarioId) {
    case "wrong-scope":
      return runWrongScopeScenario();
    case "wrong-aud":
      return runWrongAudScenario();
    // ... etc
    default:
      throw new Error(`Unknown scenario: ${scenarioId}`);
  }
}

async function runWrongScopeScenario() {
  const testToken = generateWrongScopeToken();
  const response = await callMCPWithTestToken(testToken);
  return {
    scenario: "wrong_scope",
    error_code: response.error_code || "SCOPE_MISMATCH",
    http_status: response.status || 403,
    error_description: response.error_description,
    teaching_message: "User tokens can only authorize general account operations...",
    request: { token_scopes: ["profile", "email"], endpoint: "/api/mcp/tool" },
    response: { status: response.status, error: response.error_code }
  };
}
```

```javascript
// TokenSecurityTester.jsx

import React, { useState } from "react";
import axios from "axios";

const SCENARIOS = [
  { id: "wrong-scope", label: "User Token (Wrong Scope)" },
  { id: "wrong-aud", label: "User Token (Wrong Audience)" },
  { id: "missing-act", label: "Missing Act Claim" },
  { id: "agent-token-user-endpoint", label: "Agent Token on User Endpoint" },
  { id: "expired-token", label: "Expired Token" }
];

export default function TokenSecurityTester() {
  const [selectedScenario, setSelectedScenario] = useState("wrong-scope");
  const [loading, setLoading] = useState(false);
  const [result, setResult] = useState(null);
  const [error, setError] = useState(null);

  const handleRunTest = async () => {
    setLoading(true);
    setError(null);
    setResult(null);
    
    try {
      const response = await axios.post(`/api/test/token-validation/scenario/${selectedScenario}`);
      setResult(response.data);
    } catch (err) {
      setError(err.response?.data?.error || err.message);
    } finally {
      setLoading(false);
    }
  };

  return (
    <div style={styles.container}>
      <div style={styles.warningBanner}>
        <span>⚠️ Demonstration Feature</span>
        <p>This is an educational demonstration. It is disabled in production.</p>
      </div>

      <div style={styles.controlsSection}>
        <label style={styles.label}>
          Select Test Scenario:
          <select 
            value={selectedScenario} 
            onChange={(e) => setSelectedScenario(e.target.value)}
            style={styles.select}
          >
            {SCENARIOS.map(scenario => (
              <option key={scenario.id} value={scenario.id}>
                {scenario.label}
              </option>
            ))}
          </select>
        </label>

        <button 
          onClick={handleRunTest} 
          disabled={loading}
          style={styles.runButton}
        >
          {loading ? "Running Test..." : "Run Test"}
        </button>
      </div>

      {error && (
        <div style={styles.errorBox}>
          <strong>Error:</strong> {error}
        </div>
      )}

      {result && (
        <div style={styles.resultBox}>
          <div style={styles.resultRow}>
            <strong>Scenario:</strong> {result.scenario}
          </div>
          <div style={styles.resultRow}>
            <strong>Error Code:</strong> 
            <span style={styles.errorCodeBadge}>{result.error_code}</span>
          </div>
          <div style={styles.resultRow}>
            <strong>HTTP Status:</strong> 
            <span style={styles.statusBadge}>{result.http_status}</span>
          </div>
          <div style={styles.resultRow}>
            <strong>Error Description:</strong> {result.error_description}
          </div>
          <div style={styles.teachingBox}>
            <strong>💡 Teaching Message:</strong>
            <p>{result.teaching_message}</p>
          </div>
          
          {result.request && (
            <details style={styles.details}>
              <summary>Request Details</summary>
              <pre style={styles.pre}>{JSON.stringify(result.request, null, 2)}</pre>
            </details>
          )}
          
          {result.response && (
            <details style={styles.details}>
              <summary>Response</summary>
              <pre style={styles.pre}>{JSON.stringify(result.response, null, 2)}</pre>
            </details>
          )}
        </div>
      )}
    </div>
  );
}

const styles = {
  container: {
    maxWidth: "600px",
    padding: "20px",
    backgroundColor: "#f9f9f9",
    borderRadius: "8px",
    border: "1px solid #ddd"
  },
  warningBanner: {
    backgroundColor: "#fff3cd",
    borderLeft: "4px solid #ff6b6b",
    padding: "12px",
    marginBottom: "20px",
    borderRadius: "4px",
    fontSize: "14px"
  },
  controlsSection: {
    marginBottom: "20px",
    display: "flex",
    gap: "10px",
    alignItems: "flex-end"
  },
  label: {
    display: "flex",
    flexDirection: "column",
    flex: 1,
    fontWeight: "500"
  },
  select: {
    marginTop: "8px",
    padding: "8px",
    borderRadius: "4px",
    border: "1px solid #ccc",
    fontFamily: "inherit"
  },
  runButton: {
    padding: "8px 20px",
    backgroundColor: "#007bff",
    color: "white",
    border: "none",
    borderRadius: "4px",
    cursor: "pointer",
    fontWeight: "600",
    transition: "background-color 0.2s"
  },
  errorBox: {
    backgroundColor: "#f8d7da",
    border: "1px solid #f5c6cb",
    borderRadius: "4px",
    padding: "12px",
    marginBottom: "20px",
    color: "#721c24"
  },
  resultBox: {
    backgroundColor: "white",
    border: "1px solid #ddd",
    borderRadius: "4px",
    padding: "16px"
  },
  resultRow: {
    marginBottom: "12px",
    display: "flex",
    gap: "10px",
    alignItems: "center"
  },
  errorCodeBadge: {
    backgroundColor: "#ff6b6b",
    color: "white",
    padding: "4px 8px",
    borderRadius: "3px",
    fontFamily: "monospace",
    fontSize: "12px",

  },

    overflow: "auto",
    fontSize: "12px",
    marginTop: "8px"
  }
};
```

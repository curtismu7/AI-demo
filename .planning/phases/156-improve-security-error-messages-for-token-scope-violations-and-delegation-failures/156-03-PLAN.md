```javascript
export default class ErrorDisplayService {
  // Determine how to display error based on severity
  static determineDisplay(error) {
    const errorCode = error?.error || error?.data?.error_code;
    
    const criticalErrors = [
      'TOKEN_TYPE_MISMATCH',
      'SCOPE_VIOLATION',
      'AUDIENCE_MISMATCH',
      'DELEGATION_CLAIM_MISSING',
    ];
    
    const warningErrors = [
      'RATE_LIMIT_EXCEEDED',
      'INSUFFICIENT_PERMISSIONS',
      'POLICY_VIOLATION',
    ];
    
    if (criticalErrors.includes(errorCode)) {
      return { type: 'modal', severity: 'critical', icon: '❌' };
    } else if (warningErrors.includes(errorCode)) {
      return { type: 'toast', severity: 'warning', icon: '⚠️', autoClose: 5000 };
    } else {
      return { type: 'toast', severity: 'info', icon: 'ℹ️', autoClose: 3000 };
    }
  }
  
  // Extract human-readable message from error response
  static extractMessage(error) {
    if (error?.message) return error.message;
    if (error?.data?.details?.what_failed) return error.data.details.what_failed;
    return 'An error occurred';
  }
  
  // Extract detailed error content
  static extractDetails(error) {
    const data = error?.data || error?.details || error;
    return {
      what_failed: data?.what_failed || 'Unknown error',
      why: data?.why || '',
      teaching: data?.teaching || '',
      fix: data?.fix || 'Contact support for assistance.',
      tokens_involved: data?.tokens_involved || {},
      doc_link: error?.documentation_link || '',
      error_code: error?.error || error?.error_code || 'UNKNOWN',
    };
  }
  
  // Log error to audit trail (localStorage for client-side)
  static logToAudit(error, context = {}) {
    const auditEntry = {
      timestamp: new Date().toISOString(),
      error_code: error?.error || 'UNKNOWN',
      message: this.extractMessage(error),
      details: this.extractDetails(error),
      user_email: context.user_email,
      agent_name: context.agent_name,
      endpoint: context.endpoint,
      context,
    };
    
    // Store in localStorage with max 50 entries
    let audit = JSON.parse(localStorage.getItem('error_audit_log') || '[]');
    audit.push(auditEntry);
    if (audit.length > 50) audit = audit.slice(-50);
    localStorage.setItem('error_audit_log', JSON.stringify(audit));
    
    return auditEntry;
  }
}
```

```javascript
import React from 'react';
import { toast } from 'react-toastify';

export default function ErrorToast({ error, severity = 'error' }) {
  const message = error?.message || 'An error occurred';
  const teaching = error?.details?.teaching;
  
  const content = (
    <div>
      <div style={{ fontWeight: 'bold', marginBottom: '0.5rem' }}>
        {message}
      </div>
      {teaching && (
        <div style={{ fontSize: '0.9rem', marginTop: '0.5rem', opacity: 0.9 }}>
          {teaching}
        </div>
      )}
    </div>
  );
  
  const toastConfig = {
    autoClose: 5000,
    hideProgressBar: false,
    closeOnClick: true,
    pauseOnHover: true,
  };
  
  if (severity === 'error' || severity === 'critical') {
    toast.error(content, toastConfig);
  } else if (severity === 'warning') {
    toast.warning(content, { ...toastConfig, autoClose: 6000 });
  } else {
    toast.info(content, { ...toastConfig, autoClose: 4000 });
  }
  
  return null;  // Toast is rendered by ToastContainer in App.js
}
```

   ```javascript
   import React, { useState, useEffect } from 'react';
   import './AdminErrorAuditLog.css';
   
   export default function AdminErrorAuditLog() {
     const [entries, setEntries] = useState([]);
     const [filter, setFilter] = useState('all');  // all, critical, warning, info
     const [expanded, setExpanded] = useState(null);  // Expand one entry at a time
     
     // Load from localStorage on mount
     useEffect(() => {
       const audit = JSON.parse(localStorage.getItem('error_audit_log') || '[]');
       setEntries(audit.reverse());  // Most recent first
     }, []);
     
     const filtered = entries.filter(entry => {
       if (filter === 'all') return true;
       // Map error codes to severity
       const severityMap = {
         critical: ['TOKEN_TYPE_MISMATCH', 'SCOPE_VIOLATION', ...],
         warning: ['RATE_LIMIT_EXCEEDED', 'INSUFFICIENT_PERMISSIONS', ...],
         info: ['TOKEN_EXPIRED', ...],
       };
       return severityMap[filter]?.includes(entry.error_code);
     });
     
     return (
       <div className="error-audit-log">
         <h2>Error Audit Log</h2>
         
         {/* Filter buttons */}
         <div className="audit-filters">
           {['all', 'critical', 'warning', 'info'].map(f => (
             <button
               key={f}
               className={`btn ${filter === f ? 'btn-primary' : 'btn-secondary'}`}
               onClick={() => setFilter(f)}
             >
               {f.charAt(0).toUpperCase() + f.slice(1)}
             </button>
           ))}
         </div>
         
         {/* Error entries */}
         <div className="audit-entries">
           {filtered.length === 0 ? (
             <p>No errors recorded.</p>
           ) : (
             filtered.map((entry, idx) => (
               <div key={idx} className="audit-entry">
                 {/* Row: timestamp, error code, message, expand button */}
                 <div
                   className="audit-entry__header"
                   onClick={() => setExpanded(expanded === idx ? null : idx)}
                 >
                   <span className="audit-entry__timestamp">
                     {new Date(entry.timestamp).toLocaleTimeString()}
                   </span>
                   <span className="audit-entry__code">
                     {entry.error_code}
                   </span>
                   <span className="audit-entry__message">
                     {entry.message}
                   </span>
                   <span className="audit-entry__toggle">
                     {expanded === idx ? '▼' : '▶'}
                   </span>
                 </div>
                 
                 {/* Expanded details */}
                 {expanded === idx && (
                   <div className="audit-entry__details">
                     <p><strong>What failed:</strong> {entry.details.what_failed}</p>
                     <p><strong>Why:</strong> {entry.details.why}</p>
                     <p><strong>Teaching:</strong> {entry.details.teaching}</p>
                     <p><strong>Fix:</strong> {entry.details.fix}</p>
                     {entry.user_email && (
                       <p><strong>User:</strong> {entry.user_email}</p>
                     )}
                   </div>
                 )}
               </div>
             ))
           )}
         </div>
       </div>
     );
   }
   ```

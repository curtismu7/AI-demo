# MCP Server Token Validation Points
1. **Tools/list** — Validate before returning available tools (don't show tools user can't access)
2. **Tools/call** — Validate before executing tool (primary security gate)
3. **Resources** — Validate before accessing resources

# Validation Checks Needed
1. Token type must be 'agent' (not 'user', not 'system')
2. Token scopes must include 'mcp:execute' or 'mcp:*'
3. Token must have 'act' claim (proof of delegation)
4. Token must not be expired
5. Rate limiting (< 10 requests per 60 sec)

# Error Response Format (MCP protocol)
MCP uses JsonRPC 2.0 for errors:
```json
{
  "jsonrpc": "2.0",
  "id": "request-id",
  "error": {
    "code": -32600,  // Standard JsonRpc error code
    "message": "Invalid Request",
    "data": {
      "error_code": "TOKEN_TYPE_MISMATCH",
      "details": {
        "what_failed": "...",
        "why": "...",
        "teaching": "...",
        "fix": "..."
      }
    }
  }
}
```

# MCP vs BFF Error Handling
- BFF returns HTTP status codes (403, 401, etc.)
- MCP returns JsonRPC errors with "code" field
- Both should have same educational content (what, why, teaching, fix)
- mcpErrorFormatter bridges the gap

```javascript
export default function mcpScopeValidator(requiredScopes = ['mcp:execute']) {
  return (req, res, next) => {
    const tokenScopes = req.user?.token?.scopes || [];
    
    // Check for wildcard scope or all required scopes
    const hasWildcard = tokenScopes.includes('mcp:*');
    const hasAllScopes = requiredScopes.every(scope => tokenScopes.includes(scope));
    
    if (!hasWildcard && !hasAllScopes) {
      const missingScopes = requiredScopes.filter(s => !tokenScopes.includes(s));
      return handleMcpError(res, 'SCOPE_VIOLATION', {
        required: requiredScopes,
        actual: tokenScopes,
        missing: missingScopes,
      });
    }
    
    next();
  };
}
```

```javascript
import mcpTokenValidator from '../middleware/mcpTokenValidator.js';
import mcpScopeValidator from '../middleware/mcpScopeValidator.js';

export default class ToolCallValidator {
  static validateMessage(req, toolMessage) {
    // Pre-tool-call validation:
    // 1. Token type is agent
    // 2. Scopes include mcp:execute
    // 3. 'act' claim present (delegation)
    // 4. Rate limit check
    
    // This runs BEFORE the tool executes
    // If any check fails, return error and don't execute
    
    const token = req.user?.token;
    
    // Chain all validations
    const tokenValidation = this.validateTokenType(token);
    if (tokenValidation.error) {
      return { error: tokenValidation.error, shouldExecute: false };
    }
    
    const scopeValidation = this.validateScopes(token);
    if (scopeValidation.error) {
      return { error: scopeValidation.error, shouldExecute: false };
    }
    
    const delegationValidation = this.validateDelegation(token);
    if (delegationValidation.error) {
      return { error: delegationValidation.error, shouldExecute: false };
    }
    
    const rateLimitValidation = this.validateRateLimit(req);
    if (rateLimitValidation.error) {
      return { error: rateLimitValidation.error, shouldExecute: false };
    }
    
    return { shouldExecute: true };
  }
  
  static validateTokenType(token) {
    if (token?.token_type !== 'agent') {
      return {
        error: { code: 'TOKEN_TYPE_MISMATCH', reason: 'Not an agent token' },
        valid: false,
      };
    }
    return { valid: true };
  }
  
  static validateScopes(token) {
    const requiredScopes = ['mcp:execute'];
    const actualScopes = token?.scopes || [];
    const hasRequired = requiredScopes.every(s => actualScopes.includes(s) || actualScopes.includes('mcp:*'));
    
    if (!hasRequired) {
      return {
        error: { code: 'SCOPE_VIOLATION', required: requiredScopes, actual: actualScopes },
        valid: false,
      };
    }
    return { valid: true };
  }
  
  static validateDelegation(token) {
    if (!token?.act) {
      return {
        error: { code: 'DELEGATION_CLAIM_MISSING', reason: 'No act claim found' },
        valid: false,
      };
    }
    return { valid: true };
  }
  
  static validateRateLimit(req) {

    // For now return valid
    return { valid: true };
  }
}
```

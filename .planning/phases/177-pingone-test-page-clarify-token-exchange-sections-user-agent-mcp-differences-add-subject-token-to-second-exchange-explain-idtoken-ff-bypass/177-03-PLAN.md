After the existing canExchange determination (~line 1359), add a new section:

```javascript
// --- Claim Validation: Actually perform a test exchange and validate results ---
let claimValidation = null;
if (canExchangeResult) {
  try {
    // Get a user token from session to use as subject
    const userToken = req.session?.oauthTokens?.accessToken;
    if (userToken) {
      // Perform a test exchange (Exchange 1: user-to-mcp, simplest case)
      const exchangerToken = await oauthService.getMcpExchangerToken();
      if (exchangerToken) {
        const exchangeResult = await oauthService.performTokenExchangeAs(
          exchangerToken,
          userToken,
          null, // no actor for Exchange 1
          mcpUri,
          requestedScopes.join(' ')
        );
        
        if (exchangeResult?.access_token) {
          const jwt = require('jsonwebtoken');
          const decoded = jwt.decode(exchangeResult.access_token);
          const issues = [];
          
          // Validate audience
          const actualAud = Array.isArray(decoded.aud) ? decoded.aud : [decoded.aud];
          if (!actualAud.includes(mcpUri)) {
            issues.push({
              claim: 'aud',
              expected: mcpUri,
              actual: decoded.aud,
              fix: 'MCP Resource Server audience must match PINGONE_RESOURCE_MCP_SERVER_URI'
            });
          }
          
          // Validate scopes present
          const actualScopes = (decoded.scope || '').split(' ').filter(Boolean);
          const missingScopes = requestedScopes.filter(s => !actualScopes.includes(s));
          if (missingScopes.length > 0) {
            issues.push({
              claim: 'scope',
              expected: requestedScopes,
              actual: actualScopes,
              missing: missingScopes,
              fix: 'Assign missing scopes to MCP Token Exchanger app grant'
            });
          }
          
          // Validate client_id is the exchanger app
          if (decoded.client_id && decoded.client_id !== exchangerClientId) {
            issues.push({
              claim: 'client_id',
              expected: exchangerClientId,
              actual: decoded.client_id,
              fix: 'Token client_id should match MCP Token Exchanger app'
            });
          }
          
          claimValidation = {
            tested: true,
            issues,
            allClaimsValid: issues.length === 0,
            decodedSample: {
              aud: decoded.aud,
              scope: decoded.scope,
              client_id: decoded.client_id,
              act: decoded.act || null,
              may_act: decoded.may_act || null
            }
          };
        } else {
          claimValidation = { tested: true, exchangeFailed: true, error: 'Exchange returned no access_token' };
        }
      } else {
        claimValidation = { tested: false, reason: 'Could not get MCP exchanger token' };
      }
    } else {
      claimValidation = { tested: false, reason: 'No user token in session — log in first to test claims' };
    }
  } catch (claimErr) {
    claimValidation = { tested: false, reason: claimErr.message };
  }
}
```

Add `claimValidation` to the response JSON alongside the existing fields.

Update the `canExchange` logic: keep existing value but add `claimValidation` so the frontend can show "config OK but claims have issues".

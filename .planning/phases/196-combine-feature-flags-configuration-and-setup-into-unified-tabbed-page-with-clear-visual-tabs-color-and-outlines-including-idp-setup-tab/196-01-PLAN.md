**Helper components** (add before main component):

```typescript
const ConfigValueCard: FC<{ label: string; value: string; copyable?: boolean }> = ({ label, value, copyable }) => {
  const [copied, setCopied] = useState(false);
  const handleCopy = async () => {
    try {
      await navigator.clipboard.writeText(value);
      setCopied(true);
      setTimeout(() => setCopied(false), 2000);
    } catch {
      notifyError('Copy failed — select and copy manually');
    }
  };
  return (
    <div className="cfg-value-card">
      <span className="cfg-value-label">{label}</span>
      <div className="cfg-value-row">
        <code className="cfg-value-text" title={value}>{value}</code>
        {copyable && value !== '(not configured)' && (
          <button className="cfg-copy-btn" onClick={handleCopy} title="Copy to clipboard">
            {copied ? '✓ Copied' : '📋'}
          </button>
        )}
      </div>
    </div>
  );
};

const SetupStep: FC<{ number: number; title: string; children: React.ReactNode }> = ({ number, title, children }) => (
```

**Step C: Add Feature Flags tab content rendering**

Add state for flags (inside UnifiedConfigurationPage component):

```typescript
const [flags, setFlags] = useState<any[]>([]);
const [flagsLoading, setFlagsLoading] = useState(false);
const [savingFlag, setSavingFlag] = useState<string | null>(null);

const loadFlags = useCallback(async () => {
  setFlagsLoading(true);
  try {
    const res = await bffAxios.get('/api/admin/feature-flags');
    setFlags(res.data?.flags || []);
  } catch {
    notifyError('Failed to load feature flags');
  } finally {
    setFlagsLoading(false);
  }
}, []);

useEffect(() => {
  if (activeTab === 'feature-flags' && flags.length === 0) loadFlags();
}, [activeTab, flags.length, loadFlags]);

const handleToggleFlag = async (flagId: string, newValue: boolean) => {
  setSavingFlag(flagId);
  try {
    await bffAxios.patch('/api/admin/feature-flags', { [flagId]: newValue });
    setFlags(prev => prev.map(f => f.id === flagId ? { ...f, value: newValue } : f));
    notifySuccess(`Flag ${flagId} ${newValue ? 'enabled' : 'disabled'}`);
  } catch {
    notifyError('Failed to update flag');
  } finally {
    setSavingFlag(null);
  }
};
```

```typescript
<button
  type="button"
  key={tab.id}
  className={`configuration-tab configuration-tab--${tab.color} ${activeTab === tab.id ? 'active' : ''}`}
  onClick={() => onTabChange(tab.id)}
  role="tab"
  aria-selected={activeTab === tab.id}
  aria-controls={`tabpanel-${tab.id}`}
  tabIndex={activeTab === tab.id ? 0 : -1}
  onKeyDown={(e) => {
    const tabIds = tabs.map(t => t.id);
    const idx = tabIds.indexOf(tab.id);
    if (e.key === 'ArrowRight') { e.preventDefault(); onTabChange(tabIds[(idx + 1) % tabIds.length]); }
    else if (e.key === 'ArrowLeft') { e.preventDefault(); onTabChange(tabIds[(idx - 1 + tabIds.length) % tabIds.length]); }
    else if (e.key === 'Home') { e.preventDefault(); onTabChange(tabIds[0]); }
    else if (e.key === 'End') { e.preventDefault(); onTabChange(tabIds[tabIds.length - 1]); }
  }}
>
```

1. Add handler function `handleIdTokenExchange()` in component:
```javascript
const handleIdTokenExchange = async () => {
  const config = TEST_CONFIG.idTokenExchange;
  setCurrentFlow("idTokenExchange");
  setButtonState("checking");
  
  try {
    // Call backend with 401 simulation or real MCP endpoint
    const response = await fetch(config.endpoint, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        prompt: config.testData.prompt,
        flowType: "idTokenExchange",
        // Backend will initiate 401 → auth → exchange logic
      }),
      credentials: "include"  // Include session cookie for auth
    });

    if (response.status === 401) {
      // App is redirecting to auth; wait for callback
      setButtonState("checking");
      setFlowMessage("Redirecting to authentication...");
      // Frontend should not block; backend handles redirect
      // After auth completes, backend calls MCP and returns result
      return;
    }

    if (!response.ok) {
      throw new Error(response.statusText);
    }

    const data = await response.json();
    
    // Extract token chain from response (Phase 185 integration)
    setTokenChain(data.tokenChain || {
      before: config.testData.tokenChainBefore,
      after: config.testData.tokenChainAfter,
      exchangeDetails: {
        subjectToken: "User ID token",
        actorToken: "Agent CC token",
        resultToken: "MCP token",
        actClaim: data.actClaim || "agent-user-id"
      }
    });

    setButtonState("success");
    setFlowMessage(`Success! MCP token obtained via ID token exchange.`);
    
  } catch (error) {
    setButtonState("error");
    setFlowMessage(`Error: ${error.message}`);
  }
};
```

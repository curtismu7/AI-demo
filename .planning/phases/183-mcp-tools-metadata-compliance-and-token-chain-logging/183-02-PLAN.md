**Add new method to AuditLogger class:**

/**
 * Log a token chain audit event for an MCP tool call
 * Captures complete token lineage and execution outcome per call
 * Per D-03, D-04: Full lifecycle logging with lineage tracking
 */
async logTokenChain(
  toolName: string,
  chainIndex: number,  // ordinal in session: 1st call, 2nd call, 3rd call...
  userTokenInfo: UserTokenInfo,
  exchangedTokenInfo: ExchangedTokenInfo | null,  // null if not exchanged
  context: {
    sessionId: string;
    userId?: string;
    ipAddress?: string;
    userAgent?: string;
  },
  toolExecutionStatus: 'started' | 'completed' | 'failed',
  executionResult: TokenChainExecutionResult
): Promise<void> {
  const baseEvent = this.createBaseAuditEvent(
    'token_chain',       // NEW event type
    `tool_call_${toolName}`,
    executionResult.success ? 'success' : 'failure',
    {
      sessionId: context.sessionId,
      userId: context.userId,
      ipAddress: context.ipAddress,
      userAgent: context.userAgent,
      resourceId: toolName,
      resourceType: 'transaction',  // tool calls are like transactions
      errorCode: executionResult.errorCode,
      duration: executionResult.duration
    }
  );

  const auditEvent: AuditEvent = {
    ...baseEvent,
    tokenType: exchangedTokenInfo ? 'exchanged' : 'user',
    scope: userTokenInfo.scope,  // Track scopes on this audit event
    details: {
      toolName,
      chainIndex,           // Track call order in session
      userToken: {
        sub: userTokenInfo.sub,
        scope: userTokenInfo.scope,
        issuedAt: userTokenInfo.issuedAt,
        exp: userTokenInfo.expiresAt,
        jti: userTokenInfo.tokenId
      },
      exchangedToken: exchangedTokenInfo ? {
        sub: exchangedTokenInfo.sub,
        act: exchangedTokenInfo.act,  // RFC 8693 delegation chain
        aud: exchangedTokenInfo.aud,
        scope: exchangedTokenInfo.scope,
        issuedAt: exchangedTokenInfo.issuedAt,
        exp: exchangedTokenInfo.expiresAt,
        jti: exchangedTokenInfo.tokenId
      } : null,
      toolExecutionStatus,
      result: {
        success: executionResult.success,
        errorCode: executionResult.errorCode,
        duration: executionResult.duration,
        summary: executionResult.toolResultSummary
      }
    }
  };

  await this.logger.info('Token chain audit', {
    auditEvent,
    operation: 'audit_token_chain'
  });
  await this.writeToRedis(auditEvent);
}
```

**Notes:**

- tokenType field (already in AuditEvent) distinguishes user vs. exchanged tokens

- No breaking changes to existing audit methods

const result = await this.executeSpecificTool(tool, context, agentToken);
const executionTime = Date.now() - startTime;

// NEW: Log token chain audit event (D-03, D-04)
try {
  const chainIndex = this.incrementChainIndex(session.sessionId);
  
  // Extract user token info
  const userTokenInfo = session.userTokens && session.userTokens.length > 0
    ? {
        sub: session.userId || 'unknown',
        scope: session.userTokens[0].scope?.split(' ') || [],
        issuedAt: new Date(session.userTokens[0].issuedAt).toISOString(),
        expiresAt: new Date(session.userTokens[0].issuedAt + session.userTokens[0].expiresIn * 1000).toISOString(),
        tokenId: 'user'
      }
    : {
        sub: session.userId || 'unknown',
        scope: [],
        issuedAt: new Date().toISOString(),
        expiresAt: undefined,
        tokenId: 'unknown'
      };

  // Extract exchanged token info if agent token was used
  const exchangedTokenInfo = agentToken
    ? {
        sub: 'mcp-agent',  // MCP server as delegated subject
        act: {
          iss: 'pingone',
          sub: session.userId || 'unknown'  // Original actor
        },
        scope: tool.requiredScopes || [],
        issuedAt: new Date().toISOString(),
        expiresAt: undefined,
        tokenId: 'exchange'
      }
    : null;

  // Log to AuditLogger
  await this.auditLogger.logTokenChain(
    toolName,
    chainIndex,
    userTokenInfo,
    exchangedTokenInfo,
    {
      sessionId: session.sessionId,
      userId: session.userId,
      ipAddress: context?.ipAddress,  // if available via session or request
      userAgent: context?.userAgent
    },
    'completed',  // toolExecutionStatus
    {
      success: result.success || false,
      errorCode: result.error ? 'TOOL_ERROR' : undefined,
      duration: executionTime,
      toolResultSummary: result.success
        ? `${toolName} executed successfully`
        : `${toolName} failed: ${result.error}`
    }
  );
} catch (auditError) {
  // Don't let audit failure block tool result
  this.logger.warn(`[BankingToolProvider] Failed to log token chain: ${auditError instanceof Error ? auditError.message : String(auditError)}`);
}

console.log(`[BankingToolProvider] Tool execution completed: ${toolName} (${executionTime}ms) - Success: ${result.success}`);

return result;
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| MCP Tool → AuditLogger | Tool execution details logged; ensure no user input/secrets leak |
| Token info in audit | Token claims (sub, scope, aud) logged; keep token values themselves redacted |
| Session chainIndex | Per-session call count — no sensitive operation revealed, only call ordering |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-183-04 | I (Information Disclosure) | Token info in audit | mitigate | Log token claims (sub, exp, scope) not token values. No secrets in detail fields. Audit query restricted to admin. |
| T-183-05 | I (Information Disclosure) | Tool result summary | mitigate | Summary is non-sensitive (e.g., "3 accounts", "transfer OK"). Dont log account numbers or amounts. See AuditLogger.sanitizeParams(). |
| T-183-06 | A (Availability) | AuditLogger async failure | accept | Audit logging is fire-and-forget (catch + log warning). Tool execution not blocked. Acceptable for audit non-criticality. |
| T-183-07 | A (Availability) | Redis write failure | accept | Already handled in AuditLogger.writeToRedis() — logs to stderr but doesn't throw. |

</threat_model>

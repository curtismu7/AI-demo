In the table body rendering loop (where events map over rows), add conditional:

```javascript
const renderEventRow = (event, index) => {
  if (event.eventType === 'token_chain') {
    // Special rendering for token chain events
    const detail = event.details || {};
    const userToken = detail.userToken || {};
    const exchangedToken = detail.exchangedToken;
    const result = detail.result || {};

    return (
      <tr key={`${event.eventId}-${index}`} className="token-chain-row">
        <td title={event.timestamp}>{new Date(event.timestamp).toLocaleString()}</td>
        <td className="tool-name">{detail.toolName || 'unknown'}</td>
        <td className="user-subject" title={userToken.sub}>{userToken.sub?.substring(0, 20)}...</td>
        <td className="scopes">{(userToken.scope || []).join(', ') || 'none'}</td>
        <td className="exchanged-token">
          {exchangedToken ? (
            <span title={`Sub: ${exchangedToken.sub}\nScopes: ${(exchangedToken.scope || []).join(', ')}`}>
              {exchangedToken.sub.substring(0, 15)}...
              {exchangedToken.act ? ' (delegated)' : ''}
            </span>
          ) : (
            <span className="no-exchange">Direct</span>
          )}
        </td>
        <td className={`status ${result.success ? 'success' : 'failure'}`}>
          {result.success ? '✓ Success' : '✗ Failed'}
        </td>
        <td className="duration">{result.duration || 0}ms</td>
      </tr>
    );
  }

  // Default rendering for other event types (return existing logic)
  return (
    <tr key={`${event.eventId}-${index}`}>
      {/* existing row content */}
    </tr>
  );
};
```

Then in the table body:
```jsx
<tbody>
  {filteredEvents.map((event, index) => renderEventRow(event, index))}
</tbody>
```

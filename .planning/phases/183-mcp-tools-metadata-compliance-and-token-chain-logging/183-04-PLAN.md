**Step 1: Update tokenChainService.js to add getMCPToolCalls() method**

```javascript
/**
 * Fetch MCP tool calls from audit logs for user
 * Returns array of tool call events with timestamps, status, duration
 * Lightweight view: no full token claims, just tool name + delegation flag
 * 
 * Per Phase 183 D-08, D-09: Show MCP delegation trail for all users
 */
async function getMCPToolCalls(userId) {
  try {
    // Query audit logs for tokenChain events and tool_call operations
    // This calls the MCP server's /audit endpoint via BFF
    const response = await fetch(`http://localhost:3001/audit?eventType=token_chain&userId=${encodeURIComponent(userId)}`, {
      headers: { 'Authorization': `Bearer ${process.env.AGENT_TOKEN}` }  // or use admin token
    });

    if (!response.ok) {
      console.warn('[tokenChainService] getMCPToolCalls: audit fetch failed', response.status);
      return [];
    }

    const data = await response.json();
    const events = data.events || [];

    // Extract only MCP tool call information (lightweight, user-safe view)
    return events.map(event => ({
      id: event.eventId,
      timestamp: event.timestamp,
      toolName: event.details?.toolName || 'unknown',
      status: event.details?.result?.success ? 'success' : 'failure',
      duration: event.details?.result?.duration || 0,
      chainIndex: event.details?.chainIndex || 0,
      // Do NOT include full token details (user sees delegation trail only)
      isDelegated: !!event.details?.exchangedToken,  // flag if token was exchanged
      scopes: event.details?.userToken?.scope || []   // user's scopes (not exchanged token scopes)
    })).sort((a, b) => new Date(a.timestamp) - new Date(b.timestamp));  // chronological

  } catch (error) {
    console.error('[tokenChainService] getMCPToolCalls error:', error);
    return [];
  }
}

module.exports = { getTokenChain, getCurrentTokens, synthesizeFromSession, getMCPToolCalls };
```

```javascript
export function TokenChainProvider({ children }) {
  const [sessionToken, setSessionToken] = useState(null);
  const [tokens, setTokens] = useState([]);
  const [mcpToolCalls, setMCPToolCalls] = useState([]);  // NEW: MCP delegation trail
  // ... other state

  useEffect(() => {
    const fetchTokenChain = async () => {
      try {
        const res = await fetch('/api/token-chain', { credentials: 'include' });
        const data = await res.json();
        
        setTokens(data.tokenChain || []);
        setMCPToolCalls(data.mcpToolCallsChain || []);  // NEW: Update MCP tool calls
        
        // ... process response
      } catch (error) {
        console.error('[TokenChainContext] fetchTokenChain error:', error);
        setTokens([]);
        setMCPToolCalls([]);  // NEW: Clear on error
      }
    };
    
    fetchTokenChain();
    // Poll every 10s for fresh MCP tool calls
    const pollInterval = setInterval(fetchTokenChain, 10000);
    return () => clearInterval(pollInterval);
  }, []);

  const value = {
    sessionToken,
    setSessionToken,
    tokens,
    setTokens,
    mcpToolCalls,  // NEW: export for consumers
    setMCPToolCalls
  };

  return (
    <TokenChainContext.Provider value={value}>
      {children}
    </TokenChainContext.Provider>
  );
}

export function useTokenChain() {
  const context = useContext(TokenChainContext);
  if (!context) throw new Error('useTokenChain must be used within TokenChainProvider');
  return context;
}
```

```jsx
export default function TokenChainPanel() {
  const tokenChain = useTokenChain();
  const { tokens = [], mcpToolCalls = [] } = tokenChain || {};
  const [expandedToolId, setExpandedToolId] = useState(null);
  
  return (
    <div className="token-chain-panel">
      <h3>Token & Delegation Chain</h3>
      
      {/* Existing token list section */}
      <div className="tokens-section">
        <h4>Session Tokens</h4>
        <div className="tokens-list">
          {tokens.map(token => (
            <div key={token.id} className="token-item">
              {/* existing token display */}
            </div>
          ))}
        </div>
      </div>
      
      {/* NEW: MCP Delegation Trail section */}
      <div className="mcp-trail-section">
        <h4>🔗 MCP Tool Calls ({mcpToolCalls.length})</h4>
        {mcpToolCalls.length === 0 ? (
          <p className="no-tool-calls">No MCP tool calls in this session</p>
        ) : (
          <div className="mcp-tools-list">
            {mcpToolCalls.map((toolCall, index) => (
              <div
                key={toolCall.id}
                className={`mcp-tool-call ${toolCall.status}`}
                onClick={() => setExpandedToolId(expandedToolId === toolCall.id ? null : toolCall.id)}
              >
                <div className="tool-summary">
                  <span className="tool-index">#${toolCall.chainIndex}</span>
                  <span className="tool-name">{toolCall.toolName}</span>
                  <span className={`status-badge ${toolCall.status}`}>
                    {toolCall.status === 'success' ? '✓' : '✗'}
                  </span>
                  <span className="tool-duration">{toolCall.duration}ms</span>
                  {toolCall.isDelegated && (
                    <span className="delegated-flag" title="Called with delegated token">🔀</span>
                  )}
                </div>
                <div className="tool-timestamp" title={new Date(toolCall.timestamp).toISOString()}>
                  {new Date(toolCall.timestamp).toLocaleTimeString()}
                </div>
                
                {/* Expanded detail view */}
                {expandedToolId === toolCall.id && (
                  <div className="tool-detail">
                    <div className="detail-row">
                      <span className="label">Timestamp:</span>
                      <span className="value">{new Date(toolCall.timestamp).toLocaleString()}</span>
                    </div>
                    <div className="detail-row">
                      <span className="label">CallIndex:</span>
                      <span className="value">{toolCall.chainIndex}</span>
                    </div>
                    <div className="detail-row">
                      <span className="label">Scopes:</span>
                      <span className="value">{toolCall.scopes.join(', ') || 'none'}</span>
                    </div>
                    <div className="detail-row">
                      <span className="label">Delegation:</span>
                      <span className="value">
                        {toolCall.isDelegated ? '✓ Token exchanged for MCP agent' : 'Direct user token'}
                      </span>
                    </div>
                  </div>
                )}
              </div>
            ))}
          </div>
        )}
      </div>
    </div>
  );
}
```

```javascript
const handleModelChange = async (model) => {
  try {
    const r = await fetch('/api/langchain/config', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ model }),
    });
    if (r.ok) {
      const d = await r.json();
      setStatus(prev => ({ ...prev, model: d.model }));
    }
  } catch {}
};
```

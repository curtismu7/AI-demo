    /**
     * AuthorizationAdapter.js
     *
     * JSDoc interface definition for the authorization policy engine abstraction.
     * Both SimulatedAuthorizationAdapter and PingOneAuthorizeAdapter implement this
     * interface. Custom engines (OPA, Cedar, PingOne Authorize) must satisfy this
     * contract (documented in COMPONENT.md).
     *
     * Factory: getAdapter(configStore) returns the correct implementation based on
     * ff_authorize_simulated feature flag or AUTHORIZATION_ENGINE env var.
     *
     * SECURITY NOTE: getAdapter() MUST throw if no valid engine is determinable.
     * Silent fail-open would allow unauthorized transactions.
     */
    'use strict';

    const simulatedAuthorizeService = require('../simulatedAuthorizeService');
    const configStore = require('../configStore');

    /**
     * @interface AuthorizationAdapter
     * @description Policy engine abstraction for transaction and MCP tool authorization.
     */

      /**
       * Evaluate a banking transaction for authorization.
       *
       * @param {object} context
       * @param {string} context.userId        - Subject user identifier
       * @param {number} context.amount        - Transaction amount in USD
       * @param {string} context.type          - 'transfer' | 'withdrawal' | 'deposit'
       * @param {string} [context.acr]         - Current authentication context reference
       * @param {string} [context.decisionEndpointId] - PingOne decision endpoint ID (PingOne engine only)
       * @param {string} [context.policyId]           - PingOne policy ID (PingOne engine only)
       *
       * @returns {Promise<{
       *   decision: 'PERMIT' | 'DENY' | 'INDETERMINATE',
       *   stepUpRequired: boolean,
       *   hitlRequired: boolean,
       *   path: string,
       *   decisionId: string,
       *   raw: object
       * }>}
       */

        throw new Error('AuthorizationAdapter.evaluateTransaction() not implemented');
      }

      /**
       * Evaluate whether an MCP tool call is permitted for a session.
       *
       * @param {string} toolName - MCP tool identifier (e.g. 'get_my_accounts')
       * @param {object} context
       * @param {string} context.userId          - Subject user identifier
       * @param {string} [context.tokenAudience] - JWT aud claim (space-separated if multiple)
       * @param {string} [context.actClientId]   - RFC 8693 §4.1 actor client_id
       * @param {string} [context.nestedActClientId] - Nested act chain client_id
       * @param {string} [context.mcpResourceUri]    - MCP resource URI from configStore
       * @param {string} [context.acr]           - Current authentication context reference
       *
       * @returns {Promise<{
       *   allowed: boolean,
       *   reason: string,
       *   decision: 'PERMIT' | 'DENY' | 'INDETERMINATE',
       *   hitlRequired: boolean,
       *   stepUpRequired: boolean,
       *   decisionId: string,
       *   raw: object
       * }>}
       */
      async evaluateMcpTool(toolName, context) {
        throw new Error('AuthorizationAdapter.evaluateMcpTool() not implemented');
      }
    }

    /**
     * Factory — returns the correct AuthorizationAdapter implementation.
     * Re-evaluated on each call so Config UI changes (ff_authorize_simulated toggle)
     * take effect without restart.
     *
     * @param {object} cs - configStore instance
     * @returns {AuthorizationAdapter}
     */
    function getAdapter(cs) {
      const useSimulated = simulatedAuthorizeService.isSimulatedModeEnabled(cs);
      const engineOverride = process.env.AUTHORIZATION_ENGINE;

      if (engineOverride === 'simulated' || (useSimulated && engineOverride !== 'pingone-authorize')) {
        const { SimulatedAuthorizationAdapter } = require('./SimulatedAuthorizationAdapter');
        return new SimulatedAuthorizationAdapter(cs);
      }

      if (engineOverride === 'pingone-authorize' || (!useSimulated)) {
        const { PingOneAuthorizeAdapter } = require('./PingOneAuthorizeAdapter');
        return new PingOneAuthorizeAdapter(cs);
      }

      // Defensive: should not reach here. Throw rather than fail-open.
      throw new Error(
        'AuthorizationAdapter: no valid engine configured. Set ff_authorize_simulated=true ' +
        'or AUTHORIZATION_ENGINE=pingone-authorize with required PingOne Authorize credentials.'
      );
    }

    module.exports = { AuthorizationAdapter, getAdapter };

    **Create banking_api_server/services/authorization/SimulatedAuthorizationAdapter.js:**
    ```javascript
    'use strict';
    const { AuthorizationAdapter } = require('./AuthorizationAdapter');
    const simulatedAuthorizeService = require('../simulatedAuthorizeService');

    class SimulatedAuthorizationAdapter extends AuthorizationAdapter {
      constructor(configStore) {
        super();
        this._configStore = configStore;
      }

      async evaluateTransaction(context) {
        const r = await simulatedAuthorizeService.evaluateTransaction({
          userId: context.userId,
          amount: context.amount,
          type: context.type,
          acr: context.acr,
        });
        return {
          decision: r.decision,
          stepUpRequired: !!r.stepUpRequired,
          hitlRequired: !!r.hitlRequired,
          path: r.path || 'simulated',
          decisionId: r.decisionId || '',
          raw: r.raw || {},
        };
      }

      async evaluateMcpTool(toolName, context) {
        const r = await simulatedAuthorizeService.evaluateMcpFirstTool({
          userId: context.userId,
          toolName,
          tokenAudience: context.tokenAudience || '',
          actClientId: context.actClientId || '',
          nestedActClientId: context.nestedActClientId || '',
          mcpResourceUri: context.mcpResourceUri || '',
          acr: context.acr,
        });
        const allowed = r.decision === 'PERMIT' && !r.stepUpRequired && !r.hitlRequired;
        return {
          allowed,
          reason: allowed ? 'permit' : (r.hitlRequired ? 'hitl_required' : r.stepUpRequired ? 'step_up_required' : 'deny'),
          decision: r.decision || 'INDETERMINATE',
          hitlRequired: !!r.hitlRequired,
          stepUpRequired: !!r.stepUpRequired,
          decisionId: r.decisionId || '',
          raw: {},
        };
      }
    }

    module.exports = { SimulatedAuthorizationAdapter };
    ```

    'use strict';
    const { AuthorizationAdapter } = require('./AuthorizationAdapter');
    const pingOneAuthorizeService = require('../pingOneAuthorizeService');

    class PingOneAuthorizeAdapter extends AuthorizationAdapter {
      constructor(configStore) {
        super();
        this._configStore = configStore;
      }

      async evaluateTransaction(context) {
        const r = await pingOneAuthorizeService.evaluateTransaction({
          decisionEndpointId: context.decisionEndpointId,
          policyId: context.policyId,
          userId: context.userId,
          amount: context.amount,
          type: context.type,
          acr: context.acr,
        });
        return {
          decision: r.decision,
          stepUpRequired: !!r.stepUpRequired,
          hitlRequired: false, // PingOne Authorize does not return hitlRequired in its response shape
          path: r.path || 'pingone',
          decisionId: r.decisionId || '',
          raw: r.raw || {},
        };
      }

      async evaluateMcpTool(toolName, context) {
        const r = await pingOneAuthorizeService.evaluateMcpToolDelegation({
          userId: context.userId,
          toolName,
          tokenAudience: context.tokenAudience || '',
          actClientId: context.actClientId || '',
          nestedActClientId: context.nestedActClientId || '',
          mcpResourceUri: context.mcpResourceUri || '',
          acr: context.acr,
        });
        const allowed = r.decision === 'PERMIT' && !r.stepUpRequired && !r.hitlRequired;
        return {
          allowed,
          reason: allowed ? 'permit' : (r.hitlRequired ? 'hitl_required' : r.stepUpRequired ? 'step_up_required' : 'deny'),
          decision: r.decision || 'INDETERMINATE',
          hitlRequired: !!r.hitlRequired,
          stepUpRequired: !!r.stepUpRequired,
          decisionId: r.decisionId || '',
          raw: r.raw || {},
        };
      }
    }

    module.exports = { PingOneAuthorizeAdapter };

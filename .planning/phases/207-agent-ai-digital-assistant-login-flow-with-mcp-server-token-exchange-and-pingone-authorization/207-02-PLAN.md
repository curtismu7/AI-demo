<objective>
Wire the MCP Authorize gate error codes (`mcp_step_up_required`, `mcp_authorization_denied`) into the BankingAgent's existing tool call response handler.

These errors come from `mcpToolAuthorizationService.evaluateMcpFirstToolGate()` via `server.js` when PingOne Authorize (or simulated mode) blocks the first MCP tool use per session. The BankingAgent already handles `step_up_required` and `consent_challenge_required` — this plan maps the MCP-specific codes to those same UI flows.

**Purpose:** When PingOne Authorize denies or requires step-up for first MCP tool use, the user sees appropriate UI (MFA modal or error message) instead of a generic error.

**Output:**
- `banking_api_ui/src/components/BankingAgent.js` — Updated response handler
</objective>

} else if (normalized.error === 'mcp_step_up_required') {
  // MCP first-tool gate requires MFA step-up — route to existing OTP modal
  const contextLine = normalized.error_description

  setOtpContextLine(contextLine);
  pendingOtpActionRef.current = { actionId, form };
  
  // Log decisionId for debugging/education panel
  if (normalized.decisionId) {
    console.log(`[MCP Authorize] Step-up required — decisionId: ${normalized.decisionId}, context: ${normalized.decisionContext || 'McpFirstTool'}`);
  }
  
  // Use P1MFA if available, otherwise fall back to existing step-up flow
  try {
    const apiBase = process.env.REACT_APP_API_URL || '';
    const mfaResp = await fetch(`${apiBase}/api/auth/mfa/challenge`, {
      method: 'POST',
      credentials: 'include',
      headers: { 'Content-Type': 'application/json' },
    });
    if (mfaResp.ok) {
      const { daId, devices } = await mfaResp.json();
      setP1MfaDaId(daId);
      setP1MfaDevices(devices || []);
      setP1MfaMode(true);
    }
  } catch (e) {
    console.warn('[MCP Authorize] P1MFA challenge failed, using basic OTP modal:', e.message);
  }
  
  setShowOtpModal(true);
  addMessage('assistant', `🔐 **Identity verification required**\n\n${contextLine}\n\nPlease complete the verification to continue.`, actionId);
  toast.dismiss(toastId);
  setLoading(false);
  return;

// MCP Authorize gate: access denied (PingOne Authorize or simulated mode)
} else if (normalized.error === 'mcp_authorization_denied') {

  toast.dismiss(toastId);
  setLoading(false);
  return;

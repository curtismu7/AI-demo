```javascript
// banking_api_server/services/tokenStructureValidator.js

/**
 * Validate token claims structure per RFC 8693 §3.2
 * 
 * RFC 8693 Requirements:
 * - §3.2 response must contain: sub, aud, (act if delegation), exp, scope
 * - sub = subject (user ID)
 * - act = actor (agent ID) — required when performTokenExchangeWithActor used
 * - aud = audience (resource URI) — must match requested audience
 * - exp = expiration timestamp (seconds)
 * - scope = requested scopes
 * 
 * @param {object} token - Decoded JWT payload (from jwtDecode or similar)
 * @param {string} expectedAudience - Expected aud claim (e.g., "https://mcp-server.pingdemo.com")
 * @param {string[]} expectedScopes - Expected scopes (e.g., ["banking:read", "banking:write"])
 * @param {boolean} isDelegationFlow - If true, act claim is mandatory; if false, warnings only
 * @returns {object} { valid: boolean, errors: string[], warnings: string[] }
 */
function validateTokenStructure(token, { expectedAudience, expectedScopes = [], isDelegationFlow = false } = {}) {
  const errors = [];
  const warnings = [];

  if (!token) {
    errors.push('Token is null or undefined');
    return { valid: false, errors, warnings };
  }

  // RFC 8693 §3.2 — Required: sub claim
  if (!token.sub) {
    errors.push('RFC 8693 §3.2: Missing sub claim (required — identifies subject/user)');
  } else if (typeof token.sub !== 'string' || token.sub.trim().length === 0) {
    errors.push('RFC 8693 §3.2: sub claim must be non-empty string');
  }

  // RFC 8693 §2.3 — Required: aud claim
  if (!token.aud) {
    errors.push('RFC 8693 §2.3: Missing aud claim (required — identifies resource)');
  }

  // RFC 8693 §2.3 — Audience validation
  if (expectedAudience && token.aud !== expectedAudience) {
    errors.push(
      `RFC 8693 §2.3: aud claim mismatch — expected "${expectedAudience}" but got "${token.aud}"`
    );
  }

  // RFC 8693 §2.2 — Actor claim (delegation)
  if (isDelegationFlow) {
    if (!token.act) {
      errors.push(
        'RFC 8693 §2.2: Missing act claim (required in delegation flow — identifies actor/agent)'
      );
    } else if (typeof token.act !== 'string' || token.act.trim().length === 0) {
      errors.push('RFC 8693 §2.2: act claim must be non-empty string');
    }
  } else if (!token.act) {
    warnings.push('RFC 8693 §2.2: act claim not present (warning — expected in dual-exchange)');
  }

  // Expiry validation
  if (!token.exp) {
    errors.push('RFC 8693 §3.2: Missing exp claim (required — token expiry)');
  } else {
    const now = Math.floor(Date.now() / 1000);
    if (token.exp < now) {
      errors.push(`RFC 8693 §3.2: Token expired (exp=${token.exp}, now=${now})`);
    } else if (token.exp - now < 60) {
      warnings.push(`Token expires in < 1 minute (exp=${token.exp})`);
    }
  }

  // Scope validation
  if (expectedScopes.length > 0) {
    const tokenScopes = (token.scope || '').split(' ').filter(s => s);
    const missing = expectedScopes.filter(s => !tokenScopes.includes(s));
    if (missing.length > 0) {
      errors.push(
        `Token missing required scope(s): ${missing.join(', ')} (has: ${tokenScopes.join(', ')})`
      );
    }
  } else if (!token.scope) {
    warnings.push('Token has no scope claim (may be expected depending on flow)');
  }

  return {
    valid: errors.length === 0,
    errors,
    warnings
  };
}

module.exports = { validateTokenStructure };
```

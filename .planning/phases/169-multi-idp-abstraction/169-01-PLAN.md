  <files>banking_api_server/services/oauthEndpointResolver.js</files>
  <action>
Create new file `oauthEndpointResolver.js` with centralized endpoint resolution logic:

```javascript
const configStore = require('./configStore');

/**
 * Resolves OAuth endpoints using priority:
 * 1. configStore explicit endpoint (if set)
 * 2. PingOne pattern (if env_id + region set)
 * 3. Empty string (not configured)
 */
async function getAuthorizationEndpoint() {
  const explicit = configStore.getEffective('oauth_authorization_endpoint');
  if (explicit) return explicit;
  
  const envId = configStore.getEffective('pingone_environment_id');
  const region = configStore.getEffective('pingone_region') || 'com';
  if (envId) return `https://auth.pingone.${region}/${envId}/as/authorize`;
  return '';
}

// Similar helpers for: getTokenEndpoint, getUserInfoEndpoint, getJwksUri, getIssuer

function getOAuthEndpoints() {
  return {
    authorization_endpoint: getAuthorizationEndpoint(),
    token_endpoint: getTokenEndpoint(),
    userinfo_endpoint: getUserInfoEndpoint(),
    jwks_uri: getJwksUri(),
    issuer: getIssuer(),
  };
}

module.exports = {
  getAuthorizationEndpoint,
  getTokenEndpoint,
  getUserInfoEndpoint,
  getJwksUri,
  getIssuer,
  getOAuthEndpoints,
};
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Admin Config UI → configStore | Admin can set arbitrary OAuth endpoints (including malicious); must validate URLs |
| BFF → OAuth Server | BFF trusts endpoint URLs; if poisoned, login can redirect to attacker domain |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-169-01 | Tampering | configStore OAuth endpoint fields | Mitigate | Validate endpoint URLs at startup: scheme must be `https://`, domain must be valid, no localhost in production (detect via NODE_ENV) |
| T-169-02 | Spoofing | OIDC Discovery metadata fetch | Mitigate | Verify issuer in metadata matches configured issuer; validate jwks_uri matches expected pattern |
| T-169-03 | Repudiation | Endpoint config audit trail | Accept | Config changes logged via existing configStore SET operations; sufficient for non-repudiation |
| T-169-04 | Information Disclosure | Endpoint URLs in logs | Mitigate | Do not log full endpoint URLs in production; log only domain + issuer hash |
| T-169-05 | Denial of Service | Malicious OIDC discovery | Mitigate | Timeout discovery fetch (5s max); disable discovery if it fails 3x in a row; fallback to explicit config |
| T-169-06 | Elevation of Privilege | Misconfigured issuer allows forged tokens | Mitigate | Always validate token issuer claim against configured oauth_issuer (done in tokenValidationService) |

Mitigations implemented:
- `validateOAuthEndpointUrls(endpoints)` function: checks https, valid domain, production safety
- OIDC metadata validation: issuer must match, jwks_uri must match expected pattern
- Token issuer validation: tokenValidationService always checks iss claim (existing code, no change needed)
</threat_model>

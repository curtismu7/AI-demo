const configStore = require('./configStore');
const axios = require('axios');

/**
 * Fetches OIDC discovery metadata from issuer's .well-known/openid-configuration
 * Returns metadata or throws if fetch fails or validation fails.
 */
async function fetchDiscoveryMetadata() {
  const issuer = configStore.getEffective('oauth_issuer');
  if (!issuer) return null; // No discovery if issuer not set
  
  const discoveryUrl = `${issuer}/.well-known/openid-configuration`;
  
  try {
    const response = await axios.get(discoveryUrl, { timeout: 5000 });
    const metadata = response.data;
    
    // Validate metadata
    if (!metadata.issuer) throw new Error('Missing issuer in discovery metadata');
    if (!metadata.authorization_endpoint) throw new Error('Missing authorization_endpoint');
    if (!metadata.token_endpoint) throw new Error('Missing token_endpoint');
    if (!metadata.jwks_uri) throw new Error('Missing jwks_uri');
    
    // Verify issuer matches
    const configuredIssuer = issuer.trim().replace(/\/$/, ''); // normalize
    const discoveredIssuer = metadata.issuer.trim().replace(/\/$/, '');
    if (configuredIssuer !== discoveredIssuer) {
      throw new Error(`Issuer mismatch: configured=${configuredIssuer}, discovered=${discoveredIssuer}`);
    }
    
    return metadata;
  } catch (err) {
    console.error('[oauth-discovery] Failed to fetch metadata:', err.message);
    return null; // Fallback to explicit config
  }
}

/**
 * Extracts OAuth endpoints from discovery metadata.
 * Returns { authorization_endpoint, token_endpoint, userinfo_endpoint?, jwks_uri, issuer }
 */
function extractEndpoints(metadata) {
  if (!metadata) return null;
  return {
    authorization_endpoint: metadata.authorization_endpoint,
    token_endpoint: metadata.token_endpoint,
    userinfo_endpoint: metadata.userinfo_endpoint || null,
    jwks_uri: metadata.jwks_uri,
    issuer: metadata.issuer,
  };
}

module.exports = {
  fetchDiscoveryMetadata,
  extractEndpoints,
};
```

const configStore = require('./configStore');
const discoveryService = require('./oauthDiscoveryService');

let cachedDiscoveryEndpoints = null;
let discoveryFetchedAt = 0;

async function getDiscoveredEndpoints(forceRefresh = false) {
  const now = Date.now();
  const CACHE_TTL = 3600000; // 1 hour
  
  // Use cache if available and not expired
  if (cachedDiscoveryEndpoints && !forceRefresh && (now - discoveryFetchedAt) < CACHE_TTL) {
    return cachedDiscoveryEndpoints;
  }
  
  // Fetch fresh metadata
  const metadata = await discoveryService.fetchDiscoveryMetadata();
  if (metadata) {
    cachedDiscoveryEndpoints = discoveryService.extractEndpoints(metadata);
    discoveryFetchedAt = now;
  }
  
  return cachedDiscoveryEndpoints;
}

async function getAuthorizationEndpoint() {
  // Priority:
  // 1. Explicit configStore value (oauth_authorization_endpoint)
  const explicit = configStore.getEffective('oauth_authorization_endpoint');
  if (explicit) return explicit;
  
  // 2. Discovered endpoint (if discovery enabled and available)
  const discoveryEnabled = configStore.getEffective('oauth_discovery_enabled');
  if (discoveryEnabled === 'true' || discoveryEnabled === true) {
    const discovered = await getDiscoveredEndpoints();
    if (discovered?.authorization_endpoint) return discovered.authorization_endpoint;
  }
  
  // 3. PingOne default pattern
  const envId = configStore.getEffective('pingone_environment_id');
  const region = configStore.getEffective('pingone_region') || 'com';
  if (envId) return `https://auth.pingone.${region}/${envId}/as/authorize`;
  
  return '';
}

// Similar for getTokenEndpoint, getUserInfoEndpoint, getJwksUri, getIssuer
```

  }

# OIDC Discovery Configuration

Enable automatic endpoint population from any OIDC provider's discovery metadata.

## Setup

1. Set `oauth_issuer` to your provider's issuer URL:
   - **PingOne:** `https://auth.pingone.com/YOUR_ENV_ID/as`

   - **Auth0:** `https://YOUR_TENANT.auth0.com`

   - `oauth_authorization_endpoint`
   - `oauth_token_endpoint`
   - `oauth_userinfo_endpoint`
   - `oauth_jwks_uri`

- BFF logs success/failure on startup
- If discovery fails, explicit config values are used
- If both fail, error is raised and server exits safely

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| BFF → OIDC Discovery server | BFF trusts discovery metadata; if compromised, wrong endpoints could be used |
| OIDC metadata → token validation | Metadata jwks_uri used to fetch keys; if wrong, wrong keys could validate tokens |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-169-02-01 | Tampering | Discovery metadata | Mitigate | Fetch over HTTPS only; verify issuer claim matches before using metadata |
| T-169-02-02 | Spoofing | OIDC discovery server | Mitigate | Issuer mismatch check prevents using metadata from wrong server |
| T-169-02-03 | Information Disclosure | Discovered endpoints in logs | Mitigate | Log only issuer domain, not full URLs; remove metadata from debug output in production |
| T-169-02-04 | Denial of Service | Slow discovery | Mitigate | 5-second timeout; if discovery takes >5s, fail gracefully and use explicit config |
| T-169-02-05 | Elevation of Privilege | Wrong jwks_uri | Mitigate | tokenValidationService always re-validates tokens; wrong jwks_uri would fail token verification (no privilege escalation) |

Mitigations implemented:
- Issuer validation (line 32 in discoveryService)
- HTTPS requirement (no http:// allowed)
- 5-second timeout on fetch
- Graceful fallback if discovery fails
</threat_model>

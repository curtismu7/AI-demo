In the `OAuthService` class, add a new method `performTokenExchangeFromIdToken` immediately after `performTokenExchangeWithActor`. This method is identical to `performTokenExchange` except:
- `subject_token_type` is `'urn:ietf:params:oauth:token-type:id_token'` (RFC 8693 ┬¦3, IANA token type for ID tokens)
- Log prefix is `[TokenExchange:ID-Token]`

```javascript
/**
 * RFC 8693 Token Exchange using an ID token as subject_token.
 * Used when ff_id_token_exchange is ON ŌĆö the agent is never given the user access token.
 * The ID token proves the user's identity; PingOne issues a narrowly-scoped MCP token.
 *
 * Requires PingOne token exchange policy that accepts id_token as subject_token_type.
 *
 * @param {string} idToken   - User's ID token from session (oauthTokens.idToken)
 * @param {string} audience  - Target resource URI (e.g. MCP server resource URI)
 * @param {string|string[]} scopes - Requested scopes for the exchanged token
 */
async performTokenExchangeFromIdToken(idToken, audience, scopes) {
  const scopeStr = Array.isArray(scopes) ? scopes.join(' ') : scopes;
  const body = new URLSearchParams({
    grant_type: 'urn:ietf:params:oauth:grant-type:token-exchange',
    subject_token: idToken,
    subject_token_type: 'urn:ietf:params:oauth:token-type:id_token',
    requested_token_type: 'urn:ietf:params:oauth:token-type:access_token',
    audience: audience,
    scope: scopeStr,
    client_id: this.config.clientId,
  });
  const headers = { 'Content-Type': 'application/x-www-form-urlencoded' };
  applyAdminTokenEndpointClientAuth(this.config, body, headers);
  console.log(
    '[TokenExchange:ID-Token:REQUEST] endpoint=%s client_id=%s audience=%s scope="%s"',
    this.config.tokenEndpoint,
    this.config.clientId,
    audience,
    scopeStr
  );
  try {
    const response = await axios.post(this.config.tokenEndpoint, body.toString(), { headers });
    const exchanged = response.data.access_token;
    if (!exchanged) throw new Error('ID token exchange response missing access_token');
    console.log(`[TokenExchange:ID-Token] Issued delegated token for audience=${audience} scope="${scopeStr}"`);
    return exchanged;
  } catch (error) {
    const pingoneData = error.response?.data || {};
    const httpStatus  = error.response?.status;
    console.error(
      '[TokenExchange:ID-Token:FAILED] httpStatus=%s error=%s description=%s audience=%s scope="%s"',
      httpStatus,
      pingoneData.error ?? error.message,
      pingoneData.error_description ?? '(none)',
      audience,
      scopeStr
    );
    const richErr = new Error(
      `ID token exchange failed: ${pingoneData.error_description || pingoneData.error || error.message}`
    );
    richErr.httpStatus              = httpStatus;
    richErr.pingoneError            = pingoneData.error;
    richErr.pingoneErrorDescription = pingoneData.error_description;
    richErr.pingoneErrorDetail      = pingoneData.error_detail || pingoneData.details;
    richErr.requestContext          = { audience, scope: scopeStr, client_id: this.config.clientId };
    throw richErr;
  }
}
```

```javascript
/**
 * GET /api/pingone-test/exchange-id-token-to-mcp
 * Test RFC 8693 exchange using user ID token as subject_token (ff_id_token_exchange must be ON).
 * Subject token type: urn:ietf:params:oauth:token-type:id_token
 */
router.get('/exchange-id-token-to-mcp', async (req, res) => {
  const startTime = Date.now();
  const sessionId = req.query.sessionId || 'pingone-test';

  try {
    await configStore.ensureInitialized();

    // Guard: FF must be ON
    const ffOn = configStore.getEffective('ff_id_token_exchange') === 'true'
      || configStore.getEffective('ff_id_token_exchange') === true;
    if (!ffOn) {
      const responseData = {
        success: false,
        error: 'ff_id_token_exchange is OFF. Enable it in Feature Flags to use ID token exchange.'
      };
      trackApiCall(sessionId, req, res, startTime, responseData, 'token-exchange', 'ID token ŌåÆ MCP token exchange (FF OFF)');
      return res.status(400).json(responseData);
    }

    // Guard: session must have idToken
    const oauthTokens = req.session.oauthTokens;
    if (!oauthTokens || !oauthTokens.idToken) {
      const responseData = {
        success: false,
        error: 'No ID token in session. User must log in first (openid scope required).'
      };
      trackApiCall(sessionId, req, res, startTime, responseData, 'token-exchange', 'ID token ŌåÆ MCP token exchange (no idToken)');
      return res.json(responseData);
    }

    const mcpUri = configStore.getEffective('pingone_resource_mcp_server_uri')
      || process.env.MCP_SERVER_RESOURCE_URI;
    if (!mcpUri) {
      return res.json({ success: false, error: 'MCP resource URI not configured (PINGONE_RESOURCE_MCP_SERVER_URI).' });
    }

    const scopes = (process.env.MCP_TOKEN_EXCHANGE_SCOPES || 'banking:read banking:write banking:mcp:invoke')
      .trim().split(/\s+/);

    const exchanged = await oauthService.performTokenExchangeFromIdToken(
      oauthTokens.idToken,
      mcpUri,
      scopes
    );

    const responseData = {
      success: true,
      token: exchanged.substring(0, 20) + '...',
      decoded: decodeJwtForDisplay(exchanged),
      subjectTokenType: 'id_token',
      subjectDecoded: decodeJwtForDisplay(oauthTokens.idToken),
    };
    trackApiCall(sessionId, req, res, startTime, responseData, 'token-exchange', 'ID token (RFC 8693 subject_token_type=id_token) ŌåÆ MCP token');
    res.json(responseData);
  } catch (error) {
    console.error('[PingOneTest] exchange-id-token-to-mcp error:', error.message);
    const responseData = {
      success: false,
      error: error.message,
      pingoneError: error.pingoneError,
      pingoneErrorDescription: error.pingoneErrorDescription,
    };
    trackApiCall(sessionId, req, res, startTime, responseData, 'token-exchange', 'ID token ŌåÆ MCP token exchange');
    res.json(responseData);
  }
});
```

2. In `startServer()`, AFTER the existing `await new Promise(... this.httpServer.listen(this.config.port ...))` resolves (WS server fully up on 8080 first — order matters so WS is never blocked by an h2 failure), insert:
```typescript
const h2PortRaw = process.env.MCP_HTTP2_PORT;
const h2Port = h2PortRaw ? parseInt(h2PortRaw, 10) : NaN;
if (Number.isFinite(h2Port) && h2Port > 0) {
  try {
    const http2 = await import('http2'); // dynamic import isolates the D-05 failure surface
    this.http2Server = http2.createServer({ allowHTTP1: true }, (req, res) => {
      this.handleHttpRequest(req as any, res as any); // compat layer presents IncomingMessage/ServerResponse shape — no refactor (RESEARCH A1)
    });
    this.http2Server.on('error', (err: Error) => {
      console.error(`[BankingMCPServer] h2 server error: ${err.message}`);
    });
    await new Promise<void>((resolve, reject) => {
      this.http2Server!.listen(h2Port, this.config.host, (err?: Error) => {
        if (err) reject(err); else resolve();
      });
    });
    if (this.config.enableLogging) {
      console.log(`[BankingMCPServer] HTTP/2 (h2c) listener enabled on ${this.config.host}:${h2Port} (opt-in via MCP_HTTP2_PORT)`);
    }
  } catch (err) {
    // D-05 fallback. NEVER let h2 startup break WS. Demo continues WS-only.
    console.warn(`[BankingMCPServer] ⚠️ h2 listener failed to start on :${h2Port} — continuing WS-only.`);
    console.warn(`[BankingMCPServer]    reason: ${(err as Error).message}`);
    this.http2Server = null;
  }
}
```

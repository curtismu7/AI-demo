```jsx
const BADGE_BG = { subject: "#dc2626", actor: "#2563eb", mcp: "#16a34a" };

function AuditTrailTab() {
  const ctx = useTokenChainOptional();
  const [expandedKeys, setExpandedKeys] = useState(new Set());

  if (!ctx) {
    return <div className="audit-trail-empty">Loading…</div>;
  }

  const { history, sessionTokenEvent } = ctx;

  const toggleRow = (key) =>
    setExpandedKeys((prev) => {
      const n = new Set(prev);
      n.has(key) ? n.delete(key) : n.add(key);
      return n;
    });

  // Synthetic session entry rendered at the bottom (oldest)
  const sessionEntry = sessionTokenEvent
    ? { tool: "session", timestamp: sessionTokenEvent.timestamp, events: [sessionTokenEvent] }
    : null;

  const allEntries = sessionEntry ? [...history, sessionEntry] : history;

  if (allEntries.length === 0) {
    return (
      <div className="audit-trail-empty">
        No operations yet. Use the agent to see token acquisitions here.
      </div>
    );
  }

  return (
    <div className="audit-trail-list">
      {allEntries.map((entry, i) => {
        const key = entry.timestamp ? `${entry.tool}-${entry.timestamp}` : String(i);
        const isOpen = expandedKeys.has(key);
        // Collect distinct categories from events with jwtFullDecode
        const categories = [
          ...new Set(
            entry.events
              .filter((e) => e.jwtFullDecode)
              .map((e) => deriveTokenCategory(e.label, e.id) || "unknown")
          ),
        ];
        return (
          <div key={key} className="audit-trail-row">
            <div className="audit-trail-row-header" onClick={() => toggleRow(key)}>
              <span className="audit-trail-chevron">{isOpen ? "▾" : "▸"}</span>
              <span className="audit-trail-timestamp">
                {entry.timestamp
                  ? new Date(entry.timestamp).toLocaleTimeString()
                  : "—"}
              </span>
              <span className="audit-trail-tool">{entry.tool}</span>
              <div className="audit-trail-badges">
                {categories.map((cat) => (
                  <span
                    key={cat}
                    style={{
                      background: BADGE_BG[cat] || "#475569",
                      color: "#fff",
                      borderRadius: 4,
                      padding: "2px 8px",
                      fontSize: "0.7rem",
                      fontWeight: 600,
                    }}
                  >
                    {cat}
                  </span>
                ))}
              </div>
            </div>
            {isOpen && (
              <div className="audit-trail-row-expanded">
                {entry.events
                  .filter((evt) => evt.jwtFullDecode)
                  .map((evt) => {
                    const decoded = {
                      header: evt.jwtFullDecode.header,
                      payload: evt.jwtFullDecode.claims,
                    };
                    return (
                      <DecodedTokenPanel key={evt.id || evt.label} decoded={decoded} label={evt.label} />
                    );
                  })}
              </div>
            )}
          </div>
        );
      })}
    </div>
  );
}

function TokenDecoderTab() {
  const ctx = useTokenChainOptional();

  if (!ctx) {
    return <div className="token-decoder-empty">Loading…</div>;
  }

  const eventsWithDecoded = ctx.events.filter((e) => e.jwtFullDecode);

  if (eventsWithDecoded.length === 0) {
    return (
      <div className="token-decoder-empty">
        No decoded tokens yet. Log in and run an agent action to see tokens here.
      </div>
    );
  }

  return (
    <div className="token-decoder-columns">
      {eventsWithDecoded.map((event) => {
        const decoded = {
          header: event.jwtFullDecode.header,
          payload: event.jwtFullDecode.claims,
        };
        return (
          <div key={event.id ?? event.label} className="token-decoder-column">
            <DecodedTokenPanel decoded={decoded} label={event.label} />
          </div>
        );
      })}
    </div>
  );
}
```

  requiredScopes: ['banking:accounts:read'],

  requiredScopes: ['banking:transactions:read'],

  readOnly: true,
  requiresUserAuth: false
}

**Write Operations (require auth + consent):**
```typescript
create_deposit, create_withdrawal, create_transfer: {
  requiredScopes: ['banking:transactions:write'],

  requiresUserAuth: true
}

get_sensitive_account_details: {
  requiredScopes: ['banking:sensitive:read'],
  readOnly: false,
  requiresUserAuth: true
}
```

**Public (no auth):**
```typescript
query_user_by_email: {
  requiredScopes: [],
  readOnly: false,
  requiresUserAuth: false
}
```

Component should categorize and display these with consistent card layout per ActorTokenEducation pattern.

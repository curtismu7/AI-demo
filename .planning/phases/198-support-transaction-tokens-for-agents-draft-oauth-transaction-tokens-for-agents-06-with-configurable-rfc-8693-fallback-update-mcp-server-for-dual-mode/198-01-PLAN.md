   ```javascript
   async function exchangeTokenForMcp(userToken, agentCredentials, userIdentity) {
     const config = tokenExchangeConfig;
     const primaryMode = config.mode;
     const fallbackMode = config.getFallbackMode();
     
     if (!config.isValidMode(primaryMode)) {
       console.error(`[TokenExchange] Invalid mode: ${primaryMode}`);
       // Fall back to RFC 8693
       return await exchangeTokenRfc8693(userToken, agentCredentials, userIdentity);
     }
     
     // Attempt primary mode
     let token = null;
     const logMsg = (status) => `[TokenExchange] ${primaryMode} ${status}`;
     
     try {
       if (primaryMode === "rfc_8693") {
         token = await exchangeTokenRfc8693(userToken, agentCredentials, userIdentity);
       } else if (primaryMode === "transaction_tokens") {
         token = await exchangeTokenWithTransactionTokens(userToken, agentCredentials, userIdentity);
       }
       
       if (token) {
         console.log(logMsg("succeeded"));
         return token;
       }
     } catch (err) {
       console.error(logMsg("failed:"), err.message);
     }
     
     // If primary failed and auto-fallback enabled, retry with fallback mode
     if (config.autoFallback && token === null) {
       if (config.logModeSwitches) {
         console.log(`[TokenExchange] Switching from ${primaryMode} to ${fallbackMode} (auto-fallback)`);
       }
       
       try {
         if (fallbackMode === "rfc_8693") {
           token = await exchangeTokenRfc8693(userToken, agentCredentials, userIdentity);
         } else {
           token = await exchangeTokenWithTransactionTokens(userToken, agentCredentials, userIdentity);
         }
         
         if (token) {
           console.log(`[TokenExchange] Fallback to ${fallbackMode} succeeded`);
         }
       } catch (err) {
         console.error(`[TokenExchange] Fallback to ${fallbackMode} failed:`, err.message);
       }
     }
     
     return token || null;
   }
   ```

```javascript
/**
 * POST /api/pingone-test/fix-banking-resource-server
 * Creates the banking resource server and canonical scopes in PingOne if missing.
 * Uses worker client credentials token — BFF pattern, no token returned to UI.
 * Idempotent: skips creation if RS already exists; skips existing scopes.
 */
router.post('/fix-banking-resource-server', async (req, res) => {
  const startTime = Date.now();
  const sessionId = req.query.sessionId || 'pingone-test-fix';

  // Canonical scopes to ensure exist on the banking resource server
  const CANONICAL_BANKING_SCOPES = [
    { name: 'banking:read', description: 'Read-only access to accounts, balances, and transactions' },
    { name: 'banking:write', description: 'Write access for deposits, withdrawals, transfers' },
    { name: 'banking:admin', description: 'Full administrative access' },
    { name: 'banking:sensitive', description: 'Sensitive data access (PII, account details)' },
    { name: 'banking:ai:agent', description: 'AI agent identification on banking resource tokens' },
  ];

  try {
    // 1. Acquire worker token (BFF pattern — same as verify-assets)
    let workerToken;
    try {
      workerToken = await oauthService.getAgentClientCredentialsToken();
    } catch (tokenError) {
      return res.json({
        success: false,
        error: 'Failed to obtain worker token: ' + tokenError.message +
          ' — check PINGONE_WORKER_TOKEN_CLIENT_ID and PINGONE_WORKER_TOKEN_CLIENT_SECRET in .env',
      });
    }
    managementService.initialize(workerToken);

    // 2. Check if banking resource server already exists
    const rsListResult = await managementService.getResourceServers();
    const audienceEnduser = configStore.getEffective('pingone_audience_enduser');
    const existingRS = (rsListResult.resourceServers || []).find(rs => {
      const nameLower = (rs.name || '').toLowerCase();
      const audience = rs.audience || (rs.accessControl && rs.accessControl.audience) || '';
      return nameLower.includes('banking') ||
        (audienceEnduser && audience === audienceEnduser);
    });

    let resourceServerId = existingRS && existingRS.id;
    let createdRS = false;

    // 3. Create resource server if missing
    if (!resourceServerId) {
      const audienceUri = audienceEnduser || 'https://banking-api.banking-demo.com';
      const createResult = await managementService.createResourceServer(
        'Main Banking API',
        'Banking resource server for Super Banking Demo — canonical scopes: banking:read, banking:write, banking:admin',
        audienceUri
      );
      if (!createResult || !createResult.id) {
        // Check for management API permission error
        const errMsg = (createResult && createResult.error) || 'Unknown error';
        if (errMsg.toLowerCase().includes('scope') || errMsg.toLowerCase().includes('permission') || errMsg.toLowerCase().includes('403')) {
          return res.json({
            success: false,
            error: 'Fix failed: Worker app lacks management API permissions. Grant p1:create:resourceServer scope in PingOne console.',
          });
        }
        return res.json({
          success: false,
          error: 'Fix failed: ' + errMsg + '. Check BFF logs for details.',
        });
      }
      resourceServerId = createResult.id;
      createdRS = true;
    }

    // 4. Create canonical scopes (idempotent — createScopes handles 409 per-scope)
    const scopeResult = await managementService.createScopes(resourceServerId, CANONICAL_BANKING_SCOPES);

    const responseData = {
      success: true,
      createdResourceServer: createdRS,
      resourceServerId,
      scopeResults: scopeResult,
    };

    trackApiCall(sessionId, req, res, startTime, responseData, 'pingone-test', 'Fix banking resource server: create RS + canonical scopes');
    res.json(responseData);

  } catch (error) {
    console.error('[PingOneTest] fix-banking-resource-server error:', error.message);
    const responseData = {
      success: false,
      error: error.message,
    };
    trackApiCall(sessionId, req, res, startTime, responseData, 'pingone-test', 'Fix banking resource server: create RS + canonical scopes');
    res.json(responseData);
  }
});
```

Insert this route after the last existing `router.*` block before the token-exchange route. Do not modify any other route.

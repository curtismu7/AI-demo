```javascript
// Returns { header, payload } ŌĆö safe for UI. Use parts[1] claims for TokenChainDisplay.
function decodeJwtForDisplay(token) {
  if (!token || typeof token !== 'string') { return null; }
  try {
    const parts = token.split('.');
    if (parts.length !== 3) { return null; }
    const header = JSON.parse(Buffer.from(parts[0], 'base64url').toString('utf8'));
    const payload = JSON.parse(Buffer.from(parts[1], 'base64url').toString('utf8'));
    return { header, payload };
  } catch (_e) { return null; }
}
```

```javascript
/**
 * Build a token event object compatible with the Token Chain UI (TokenChainDisplay.js).
 * Shape matches agentMcpTokenService.buildTokenEvent ŌĆö deviations will silently render nothing.
 *
 * @param {string} id       - Unique event ID (e.g. 'exchange-1-subject-token')
 * @param {string} label    - Human label (e.g. 'User Token (Subject)')
 * @param {string} status   - 'active' | 'exchanged' | 'failed' | 'skipped'
 * @param {object|null} decoded - Output of decodeJwtForDisplay(token) i.e. { header, payload }
 * @param {string} explanation  - Human-readable description of what happened
 * @param {object} [extra]      - Extra fields (error, rfc, etc.)
 * @returns {object}
 */
function buildExchangeTokenEvent(id, label, status, decoded, explanation, extra = {}) {
  const header = decoded?.header ?? null;
  const payload = decoded?.payload ?? null;
  // Sanitize claims ŌĆö only the fields TokenChainDisplay renders
  const claims = payload ? (() => {
    const c = {};
    ['sub', 'aud', 'scope', 'iss', 'exp', 'iat', 'nbf', 'may_act', 'act', 'client_id', 'azp'].forEach(k => {
      if (payload[k] != null) c[k] = payload[k];
    });
    return c;
  })() : null;
  return {
    id,
    label,
    status,
    timestamp: new Date().toISOString(),
    alg: header?.alg ?? null,
    claims,
    explanation,
    jwtFullDecode: header && payload ? { header, claims: payload } : null,
    error: extra.error ?? null,
    ...extra,
  };
}
```

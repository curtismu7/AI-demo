**Part C — BrandLogo.js null guard:** Replace the current `return (...)` block with a null guard
so an empty logoPath does not render a broken img element:

```javascript
export default function BrandLogo({
  className = "",
  style = {},
  height = 40,
  width = 40,
}) {
  const { preset } = useIndustryBranding();
  if (!preset.logoPath) return null;
  return (
    <img
      src={preset.logoPath}
      alt=""
      className={className}
      style={{ height, width, objectFit: "contain", ...style }}
    />
  );
}
```

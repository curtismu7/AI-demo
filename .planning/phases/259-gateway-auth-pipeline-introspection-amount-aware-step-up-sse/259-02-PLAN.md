```javascript
/** Renders the gateway RFC 7662 introspection card. */
function GatewayIntrospectionEduBox({ event }) {
  if (event.id !== 'gw-introspection') return <NotApplicableNote rfc="RFC 7662 §2.2" rfcDesc="Gateway active-token introspection — the MCP gateway asks PingOne in real time whether the inbound delegated token is still active before evaluating any policy." />;
  const result = event.extra?.introspectionResult || {};
  const status = event.eventStatus || 'skipped';
  const meta = {
    active:  { icon: '✅', cls: 'tcd-edu-box--ok',      label: 'Token Active' },
    failed:  { icon: '❌', cls: 'tcd-edu-box--error',   label: 'Token Inactive — Request Blocked' },
    skipped: { icon: '⏭', cls: 'tcd-edu-box--neutral', label: 'Introspection Not Configured' },
  }[status] || { icon: '⏭', cls: 'tcd-edu-box--neutral', label: status };

  return (
    <div className={`tcd-edu-box ${meta.cls}`}>
      <div className="tcd-edu-box-hd">
        <span className="tcd-edu-icon">{meta.icon}</span>
        <strong>Gateway — RFC 7662 Active-Token Introspection — {meta.label}</strong>
        <span className="tcd-edu-ref">RFC 7662 §2.2</span>
      </div>
      <div className="tcd-edu-body">
        {status === 'active' && (
          <>
            <p>The MCP Gateway confirmed the delegated token is <strong>active</strong> at PingOne before evaluating any authorization policy. This catches revoked tokens that are not yet expired — JWKS alone cannot detect revocation.</p>
            <ul>
              {result.sub   && <li><code>sub</code>: <strong>{result.sub}</strong></li>}
              {result.scope && <li><code>scope</code>: <code>{result.scope}</code></li>}
              {result.exp   && <li><code>exp</code>: {new Date(result.exp * 1000).toLocaleTimeString()}</li>}
            </ul>
            <p className="tcd-edu-detail">RFC 7662 §2.2: <code>active: true</code> means the token has not been revoked and is within its validity window at the authorization server.</p>
          </>
        )}
        {status === 'failed' && (
          <><p>PingOne returned <code>active: false</code>. The gateway blocked the request immediately — no policy evaluation or token exchange was attempted.</p>
          <div className="tcd-edu-fix"><strong>Fix:</strong> Sign out and sign in again to get a fresh session token.</div></>
        )}
        {status === 'skipped' && (
          <p><code>GW_INTROSPECTION_ENDPOINT</code> is not set. The gateway skipped revocation checking and relies on JWT expiry only. Set the endpoint for zero-trust validation at the gateway layer.</p>
        )}
      </div>
    </div>
  );
}

/** Renders the gateway PingOne Authorize decision card. */
function GatewayAuthorizeEduBox({ event }) {
  if (event.id !== 'gw-authorize') return <NotApplicableNote rfc="PingOne Authorize" rfcDesc="The MCP gateway evaluates a per-request policy in PingOne Authorize (DaVinci) covering tool access, scope sufficiency, and transaction thresholds — before forwarding to the MCP server." />;
  const decision = event.extra?.decision || event.eventStatus || 'unknown';
  const reason = event.extra?.reason || '';
  const decisionMeta = {
    PERMIT:         { icon: '✅', cls: 'tcd-edu-box--ok',      label: 'PERMIT' },
    DENY:           { icon: '❌', cls: 'tcd-edu-box--error',   label: 'DENY — Request Blocked' },
    INDETERMINATE:  { icon: '⚠️', cls: 'tcd-edu-box--neutral', label: 'INDETERMINATE — Step-Up Required' },
  }[decision] || { icon: '❓', cls: 'tcd-edu-box--neutral', label: decision };

  return (
    <div className={`tcd-edu-box ${decisionMeta.cls}`}>
      <div className="tcd-edu-box-hd">
        <span className="tcd-edu-icon">{decisionMeta.icon}</span>
        <strong>Gateway — PingOne Authorize Decision: {decisionMeta.label}</strong>
        <span className="tcd-edu-ref">PingOne Authorize · DaVinci</span>
      </div>
      <div className="tcd-edu-body">
        {decision === 'PERMIT' && (
          <><p>PingOne Authorize evaluated the request context — tool name, token scopes, transaction amount, and delegation chain — and issued a <strong>PERMIT</strong>. The gateway proceeded to token exchange.</p>
          <p className="tcd-edu-detail">The Authorize policy can enforce rules that are impossible to express in static scopes: per-transaction dollar thresholds, time-of-day restrictions, risk scores, and delegation depth limits.</p></>
        )}
        {decision === 'DENY' && (
          <><p>PingOne Authorize <strong>denied</strong> the request. The tool call was blocked at the gateway — the MCP server never received it.</p>
          {reason && <p><strong>Reason:</strong> {reason}</p>}
          <div className="tcd-edu-fix"><strong>Why:</strong> The token may lack a required scope, the tool may not be in the allow-list, or a policy constraint was violated.</div></>
        )}
        {decision === 'INDETERMINATE' && (
          <><p>The transaction requires <strong>step-up authentication</strong>. PingOne Authorize returned INDETERMINATE because the transaction amount exceeds the configured threshold and the current token does not carry a sufficient ACR claim.</p>
          {reason && <p><strong>Reason:</strong> {reason}</p>}
          <p className="tcd-edu-detail">The gateway returned HTTP 428 to the BFF, which triggered the step-up MFA flow. After the user completes MFA, a new elevated token is issued and the transaction is retried automatically.</p></>
        )}
      </div>
    </div>
  );
}
```
